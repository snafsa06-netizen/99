local SaveFile = "acrocan0_joystick_settings.json"
local _G_settings = {
    CustomJoystick = false,
    JoySize = 140,
    JoyPosX = 20,
    JoyPosY = -70
}

pcall(function()
    if readfile and isfile and isfile(SaveFile) then
        local data = HttpService:JSONDecode(readfile(SaveFile))
        for k,v in pairs(data) do
            _G_settings[k]=v
        end
    end
end)

local function SaveSettings()
    pcall(function()
        if writefile then
            writefile(SaveFile,HttpService:JSONEncode(_G_settings))
        end
    end)
end

local GrayGui=nil
local activeTouch=nil
local Controls=nil

local function SetMoveVector(vec)
    pcall(function()
        if not Controls then
            Controls = require(lp.PlayerScripts:WaitForChild("PlayerModule")):GetControls()
        end
        local c = Controls:GetActiveController()
        if c then
            c.moveVector = vec
        end
    end)
end

local function stopMoving()
    activeTouch=nil
    SetMoveVector(Vector3.new(0,0,0))
    if GrayGui and GrayGui:FindFirstChild("TouchBlocker") then
        local stick = GrayGui.TouchBlocker.MainFrame.Stick
        local half = _G_settings.JoySize/2
        stick:TweenPosition(UDim2.new(0.5,-half/2,0.5,-half/2),"Out","Back",0.1)
    end
end

local function ApplyJoystickSettings()
    if GrayGui and GrayGui:FindFirstChild("TouchBlocker") then
        local main = GrayGui.TouchBlocker.MainFrame
        main.Size = UDim2.new(0,_G_settings.JoySize,0,_G_settings.JoySize)
        main.Position = UDim2.new(0,_G_settings.JoyPosX,0.5,_G_settings.JoyPosY)
        local stick = main.Stick
        local half = _G_settings.JoySize/2
        stick.Size = UDim2.new(0,half,0,half)
        if not activeTouch then
            stick.Position = UDim2.new(0.5,-half/2,0.5,-half/2)
        end
    end
end

local function setupGrayJoy()
    local sg = Instance.new("ScreenGui",pgui)
    sg.Name = "Gray_Movement"
    sg.ResetOnSpawn = false
    sg.IgnoreGuiInset = true
    sg.DisplayOrder = -100
    sg.Enabled = false

    local touchBlocker = Instance.new("Frame",sg)
    touchBlocker.Name = "TouchBlocker"
    touchBlocker.Size = UDim2.new(0,400,0,220)
    touchBlocker.Position = UDim2.new(0,0,1,-220)
    touchBlocker.BackgroundTransparency = 1
    touchBlocker.Active = true

    local main = Instance.new("Frame",touchBlocker)
    main.Name = "MainFrame"
    main.BackgroundColor3 = Color3.fromRGB(0,0,0)
    main.BackgroundTransparency = 0.4
    main.Active = true
    Instance.new("UICorner",main).CornerRadius = UDim.new(1,0)
    local ms = Instance.new("UIStroke",main)
    ms.Color = Color3.fromRGB(150,0,0)
    ms.Thickness = 2

    local stick = Instance.new("Frame",main)
    stick.Name = "Stick"
    stick.BackgroundColor3 = Color3.fromRGB(180,0,0)
    stick.BackgroundTransparency = 0.25
    Instance.new("UICorner",stick).CornerRadius = UDim.new(1,0)
    local ss = Instance.new("UIStroke",stick)
    ss.Color = Color3.fromRGB(255,40,40)
    ss.Thickness = 1.5

    GrayGui = sg
    ApplyJoystickSettings()

    main.InputBegan:Connect(function(input)
        if input.UserInputType==Enum.UserInputType.Touch or input.UserInputType==Enum.UserInputType.MouseButton1 then
            activeTouch=input
            local center = main.AbsolutePosition + (main.AbsoluteSize/2)
            local diff = Vector2.new(input.Position.X-center.X,input.Position.Y-center.Y)
            local limit = main.AbsoluteSize.X/2
            local dist = math.min(diff.Magnitude,limit)
            if dist>0.05 then
                local dir = diff.Unit
                SetMoveVector(Vector3.new(dir.X*(dist/limit),0,dir.Y*(dist/limit)))
            end
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if input==activeTouch and sg.Enabled then
            local center = main.AbsolutePosition + (main.AbsoluteSize/2)
            local diff = Vector2.new(input.Position.X-center.X,input.Position.Y-center.Y)
            local limit = main.AbsoluteSize.X/2
            local dist = math.min(diff.Magnitude,limit)
            local dir = diff.Unit
            local half = _G_settings.JoySize/2
            stick.Position = UDim2.new(0.5,(dir.X*dist)-half/2,0.5,(dir.Y*dist)-half/2)
            if dist>0.05 then
                SetMoveVector(Vector3.new(dir.X*(dist/limit),0,dir.Y*(dist/limit)))
            else
                SetMoveVector(Vector3.new(0,0,0))
            end
        end
    end)

    UserInputService.InputEnded:Connect(function(input)
        if input==activeTouch then
            stopMoving()
        end
    end)

    return sg
end

local function UpdateGrayMovementState()
    local touchGui = pgui:FindFirstChild("TouchGui")

    local function toggleDefaultThumbstick(visible)
        if touchGui then
            local frame = touchGui:FindFirstChild("TouchControlFrame")
            if frame then
                for _,v in pairs(frame:GetChildren()) do
                    if string.find(v.Name,"Thumbstick") then
                        for _,d in pairs(v:GetDescendants()) do
                            if d:IsA("ImageLabel") or d:IsA("ImageButton") then
                                d.ImageTransparency = visible and 0 or 1
                            end
                        end
                    end
                end
            end
        end
    end

    if _G_settings.CustomJoystick then
        toggleDefaultThumbstick(false)
        if not GrayGui then GrayGui = setupGrayJoy() end
        GrayGui.Enabled = true
        ApplyJoystickSettings()
    else
        toggleDefaultThumbstick(true)
        if GrayGui then
            GrayGui.Enabled = false
            stopMoving()
        end
    end
end

local dragging=false
local dragInput
local dragStart
local startPos

MainFrame.InputBegan:Connect(function(input)
    if input.UserInputType==Enum.UserInputType.MouseButton1 or input.UserInputType==Enum.UserInputType.Touch then
        dragging=true
        dragStart=input.Position
        startPos=MainFrame.Position
        input.Changed:Connect(function()
            if input.UserInputState==Enum.UserInputState.End then
                dragging=false
            end
        end)
    end
end)

MainFrame.InputChanged:Connect(function(input)
    if input.UserInputType==Enum.UserInputType.MouseMovement or input.UserInputType==Enum.UserInputType.Touch then
        dragInput=input
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if input==dragInput and dragging then
        local delta=input.Position-dragStart
        MainFrame.Position=UDim2.new(startPos.X.Scale,startPos.X.Offset+delta.X,startPos.Y.Scale,startPos.Y.Offset+delta.Y)
    end
end)
