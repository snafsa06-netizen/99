local GrayGui = nil
local activeTouch = nil

local function SetMoveVector(vec)
    pcall(function()
        if not Controls then Controls = require(lp:WaitForChild("PlayerScripts"):WaitForChild("PlayerModule")):GetControls() end
        local activeController = Controls:GetActiveController()
        if activeController then activeController.moveVector = vec end

local function ApplyJoystickSettings()
    if GrayGui and GrayGui:FindFirstChild("TouchBlocker") then
        local main = GrayGui.TouchBlocker.MainFrame
        main.Size = UDim2.new(0, _G_settings.JoySize, 0, _G_settings.JoySize)
        main.Position = UDim2.new(0, _G_settings.JoyPosX, 0.5, _G_settings.JoyPosY)
        local stick = main.Stick
        local half = _G_settings.JoySize / 2
        stick.Size = UDim2.new(0, half, 0, half)
        if not activeTouch then
            stick.Position = UDim2.new(0.5, -half/2, 0.5, -half/2)
        end
    end
end

local function stopMoving()
    activeTouch = nil
    moveDir = Vector3.new(0,0,0)
    if GrayGui and GrayGui:FindFirstChild("TouchBlocker") then
        local stick = GrayGui.TouchBlocker.MainFrame.Stick
        local half = _G_settings.JoySize / 2
        stick:TweenPosition(UDim2.new(0.5, -half/2, 0.5, -half/2), "Out", "Back", 0.1)
    end
    SetMoveVector(Vector3.new(0, 0, 0))
end

local function setupGrayJoy()
    local sg = Instance.new("ScreenGui", pgui)
    sg.Name = "Gray_Movement"
    sg.DisplayOrder = -100
    sg.Enabled = false
    sg.ResetOnSpawn = false 
    
    local touchBlocker = Instance.new("Frame", sg)
    touchBlocker.Name = "TouchBlocker"
    touchBlocker.Size = UDim2.new(0, 400, 0, 180)
    touchBlocker.Position = UDim2.new(0, 10, 1, -200)
    touchBlocker.BackgroundTransparency = 1
    touchBlocker.ZIndex = -10
    touchBlocker.Active = true 
    
    local main = Instance.new("Frame", touchBlocker)
    main.Name = "MainFrame"
    main.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
    main.BackgroundTransparency = 0.85
    main.ZIndex = -9
    main.Active = true 
    Instance.new("UICorner", main).CornerRadius = UDim.new(1, 0)
    local mainStroke = Instance.new("UIStroke", main)
    mainStroke.Color = Color3.fromRGB(150, 0, 0)
    
    local stick = Instance.new("Frame", main)
    stick.Name = "Stick"
    stick.BackgroundColor3 = Color3.fromRGB(180, 0, 0)
    stick.BackgroundTransparency = 0.6
    stick.ZIndex = -8
    Instance.new("UICorner", stick).CornerRadius = UDim.new(1, 0)

    GrayGui = sg
    ApplyJoystickSettings()

    main.InputBegan:Connect(function(input)
        if (input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1) then
            activeTouch = input
            local center = main.AbsolutePosition + (main.AbsoluteSize / 2)
            local diff = Vector2.new(input.Position.X - center.X, input.Position.Y - center.Y)
            local limit = main.AbsoluteSize.X / 2
            local dist = math.min(diff.Magnitude, limit)
            if dist > 0.05 then
                local dir = diff.Unit
                SetMoveVector(Vector3.new(dir.X * (dist/limit), 0, dir.Y * (dist/limit)))
            end
        end
    end)
    UserInputService.InputChanged:Connect(function(input)
        if input == activeTouch and sg.Enabled then
            local center = main.AbsolutePosition + (main.AbsoluteSize / 2)
            local diff = Vector2.new(input.Position.X - center.X, input.Position.Y - center.Y)
            local limit = main.AbsoluteSize.X / 2
            local dist = math.min(diff.Magnitude, limit)
            local dir = diff.Unit
            local half = _G_settings.JoySize / 2
            stick.Position = UDim2.new(0.5, (dir.X * dist) - half/2, 0.5, (dir.Y * dist) - half/2)
            if dist > 0.05 then SetMoveVector(Vector3.new(dir.X * (dist/limit), 0, dir.Y * (dist/limit))) else SetMoveVector(Vector3.new(0, 0, 0)) end
        end
    end)
    UserInputService.InputEnded:Connect(function(input) if input == activeTouch then stopMoving() end end)
    UserInputService.WindowFocusReleased:Connect(stopMoving)
    return sg
end

local function UpdateGrayMovementState()
    local shouldBeActive = _G_settings.CustomJoystick
    local touchGui = pgui:FindFirstChild("TouchGui")
    local function toggleDefaultThumbstick(visible)
        if touchGui then
            local frame = touchGui:FindFirstChild("TouchControlFrame")
            if frame then
                for _, v in pairs(frame:GetChildren()) do
                    if string.find(v.Name, "Thumbstick") then
                        for _, desc in pairs(v:GetDescendants()) do if desc:IsA("ImageLabel") or desc:IsA("ImageButton") then desc.ImageTransparency = visible and 0 or 1 end end
                        v.BackgroundTransparency = 1
                    end
                end
            end
        end
    end
    if shouldBeActive then
        toggleDefaultThumbstick(false)
        if not GrayGui then GrayGui = setupGrayJoy() end
        GrayGui.Enabled = true
        ApplyJoystickSettings()
    else
        toggleDefaultThumbstick(true)
        if GrayGui and GrayGui.Enabled then GrayGui.Enabled = false; stopMoving() end
    end
end

-- [[ NEW Parry Systems Engine (Ultra Fast) ]]
-- ==========================================
local isParrying = false
local antiParryCooldown = false

if isMainGame then
    local animationCache = {}
    local AnimKeywords = {'slash', 'swing', 'sword', 'hit', 'punch', 'combo', 'light', 'heavy', 'attack', 'strike', 'm1'}
    local DefenseNames = {'block', 'parry', 'guard', 'defend', 'shield'}

    local function getAnimInfoCache(id)
        if animationCache[id] then return animationCache[id] end
        task.spawn(function()
            local success, info = pcall(function() return Marketplace:GetProductInfo(id) end)
            if success and info then 
                animationCache[id] = info.Name:lower()
            end
        end)
        return ""
    end

    local function performParry()
        if isParrying then return end 
        isParrying = true
        
        if keypress then 
            keypress(0x46) -- F key
            task.delay(0.015, function()
                keyrelease(0x46)
                isParrying = false
            end)
        end 
    end

    local function triggerAntiParry()
        if antiParryCooldown then return end
        antiParryCooldown = true

        if keypress then
            keypress(0x31) -- 1 key
            task.delay(0.015, function() keyrelease(0x31) end)
            
            task.delay(0.4, function() 
                keypress(0x31)
                task.delay(0.015, function() keyrelease(0x31) end)
                
                task.delay(0.05, function()
                    antiParryCooldown = false
                end)
            end)
        end
    end

    local function setupParrySystems(playerToMonitor)
        local function connectCharacter(char)
            local humanoid = char:WaitForChild("Humanoid", 10)
            local animator = humanoid:WaitForChild("Animator", 10) or humanoid
            if not animator then return end
            
            animator.AnimationPlayed:Connect(function(track)
                if not _G_settings.AutoParry and not _G_settings.AntiParry then return end
                
                local myChar = lp.Character
                local myRoot = myChar and myChar:FindFirstChild("HumanoidRootPart")
                local targetRoot = char:FindFirstChild("HumanoidRootPart")
                
                if not myRoot or not targetRoot then return end
                
                local dist = (targetRoot.Position - myRoot.Position).Magnitude
                if dist > math.max(_G_settings.AutoParryDist, _G_settings.AntiParryDist) then return end

                local localAnimName = track.Name or (track.Animation and track.Animation.Name) or ""
                local lowerLocalName = localAnimName:lower()
                
                local isAttack = false
                local isDefense = false

                for i = 1, #AnimKeywords do
                    if string.find(lowerLocalName, AnimKeywords[i]) then 
                        isAttack = true
                        break 
                    end 
                end

                if not isAttack then
                    for i = 1, #DefenseNames do
                        if string.find(lowerLocalName, DefenseNames[i]) then
                            isDefense = true
                            break
                        end
                    end
                end

                if isAttack and _G_settings.AutoParry and dist <= _G_settings.AutoParryDist then
                    performParry()
                    return 
                end

                if isDefense and _G_settings.AntiParry and dist <= _G_settings.AntiParryDist then
                    triggerAntiParry()
                    return
                end

                local animId = track.Animation.AnimationId
                local idNum = tonumber(animId:match("%d+"))
                if not idNum then return end

                local name = animationCache[idNum]
                if name then
                    if _G_settings.AutoParry and dist <= _G_settings.AutoParryDist then
                        for i = 1, #AnimKeywords do
                            if string.find(name, AnimKeywords[i]) then 
                                performParry()
                                return
                            end 
                        end
                    end
                    
                    if _G_settings.AntiParry and dist <= _G_settings.AntiParryDist then
                        for i = 1, #DefenseNames do
                            if string.find(name, DefenseNames[i]) then
                                triggerAntiParry()
                                return
                            end
                        end
                    end
                else
                    getAnimInfoCache(idNum)
                end
            end)
        end
        if playerToMonitor.Character then connectCharacter(playerToMonitor.Character) end
        playerToMonitor.CharacterAdded:Connect(connectCharacter)
    end

    for _, player in pairs(game:GetService("Players"):GetPlayers()) do 
        if player ~= lp then 
            setupParrySystems(player) 
        end 
    end
    game:GetService("Players").PlayerAdded:Connect(setupParrySystems)
end

    dodgeSound.SoundId = "rbxassetid://134012322" 
    dodgeSound.Volume = 2
    local cachedKunais = {}
    local last_scan = 0

    RunService.Heartbeat:Connect(function()
        if not _G_settings.KunaiDodge then return end
        local char = lp.Character
        local hrp = char and char:FindFirstChild("HumanoidRootPart")
        if not hrp then return end

        if dodgeSound.Parent ~= hrp then
            dodgeSound.Parent = hrp
        end

        local now = tick()
        if now - last_scan > 0.5 then
            last_scan = now
            cachedKunais = {}
            local kunaiFolder = workspace:FindFirstChild("Kunai")
            if kunaiFolder then
                local projParts = kunaiFolder:FindFirstChild("ProjectileParts")
                if projParts then
                    for _, k in ipairs(projParts:GetChildren()) do
                        if k.Name == "Kunai" and k:IsA("BasePart") then table.insert(cachedKunais, k) end
                    end
                end
            end
            local junkFolder = workspace:FindFirstChild("EffectsJunk")
            if junkFolder then
                local cache = junkFolder:FindFirstChild("PartCache")
                local searchTarget = cache or junkFolder
                for _, k in ipairs(searchTarget:GetChildren()) do
                    if k.Name == "Kunai" and k:IsA("BasePart") then table.insert(cachedKunais, k) end
                end
            end
        end

        for _, obj in ipairs(cachedKunais) do
            if obj and obj.Parent and not obj:IsDescendantOf(char) then
                if obj.Position.Y > 10000 or obj.Position.Y < -10000 then continue end
                local dist = (obj.Position - hrp.Position).Magnitude
                if dist < DETECTION_DIST then
                    local lookDir = obj.CFrame.LookVector
                    local dirToMe = (hrp.Position - obj.Position).Unit
                    if lookDir:Dot(dirToMe) > 0.8 then
                        if now - last_dodge > cooldown then
                            last_dodge = now
                            local side = hrp.CFrame.RightVector * (math.random() > 0.5 and DODGE_DISTANCE or -DODGE_DISTANCE)
                            hrp.CFrame = hrp.CFrame + side
                            dodgeSound:Play()
                        end
                    end
                end
            end
        end
    end)
end

-- Anti-Grenade
local function StartAntiGrenade()
    StopAnti("Grenade")

    colorCorrection.Name = "Acrocan0_GrenadeFilter"
    colorCorrection.TintColor = Color3.fromRGB(255, 255, 255)
    colorCorrection.Parent = Lighting

    local activeGrenades = {}
    local connections = {}

    local function createSafetyWall(grenade)
        task.wait(2)
        if not grenade or not grenade.Parent then return end

        local shieldRadius = 20
        local activationRange = 60
        local isVisualsActive = false
        local sphere, billboard, textLabel, connection = nil, nil, nil, nil

        local function cleanupVisuals()
            if not isVisualsActive then return end
            isVisualsActive = false
            activeGrenades[grenade] = nil
            if billboard then billboard:Destroy(); billboard = nil end
            if sphere then sphere:Destroy(); sphere = nil end
        end

        local function createVisuals()
            if isVisualsActive then return end
            isVisualsActive = true
            activeGrenades[grenade] = shieldRadius

            sphere = Instance.new("SphereHandleAdornment")
            sphere.Radius = shieldRadius
            sphere.Color3 = Color3.fromRGB(255, 0, 0)
            sphere.Transparency = 1
            sphere.Adornee = grenade
            sphere.AlwaysOnTop = false
            sphere.Parent = grenade
            TweenService:Create(sphere, TweenInfo.new(0.5), {Transparency = 0.75}):Play()

            billboard = Instance.new("BillboardGui")
            billboard.Size = UDim2.new(0, 200, 0, 50)
            billboard.Adornee = grenade
            billboard.AlwaysOnTop = true
            billboard.ExtentsOffset = Vector3.new(0, 4, 0)
            
            textLabel = Instance.new("TextLabel")
            textLabel.Parent = billboard
            textLabel.BackgroundTransparency = 1
            textLabel.Size = UDim2.new(1, 0, 1, 0)
            textLabel.Text = "Acrocan0 Shield"
            textLabel.TextColor3 = Color3.fromRGB(255, 0, 0)
            textLabel.TextScaled = true
            textLabel.Font = Enum.Font.GothamBold
            textLabel.TextTransparency = 1
            textLabel.TextStrokeTransparency = 1
            textLabel.TextStrokeColor3 = Color3.fromRGB(0, 0, 0)
            billboard.Parent = grenade
            
            TweenService:Create(textLabel, TweenInfo.new(0.5), {TextTransparency = 0, TextStrokeTransparency = 0.3}):Play()
            local pulse = TweenService:Create(textLabel, TweenInfo.new(0.5, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut, -1, true), {TextTransparency = 0.6})
            pulse:Play()

            if not grenade or not grenade.Parent then
                cleanupVisuals()
                if connection then connection:Disconnect() end
                return
            end
            local char = lp.Character
            if char and char:FindFirstChild("HumanoidRootPart") then
                local root = char.HumanoidRootPart
                local distance = (root.Position - grenade.Position).Magnitude
                if distance <= activationRange then
                    createVisuals()
                    if distance <= shieldRadius + 0.5 then
                        local pushDir = (root.Position - grenade.Position).Unit
                        root.AssemblyLinearVelocity = Vector3.new(pushDir.X * 145, root.AssemblyLinearVelocity.Y, pushDir.Z * 145)
                    end
                else
                    cleanupVisuals()
                end
            end
        end)
        table.insert(connections, connection)
    end

    local renderConn = RunService.RenderStepped:Connect(function()
        local char = lp.Character
        if not char or not char:FindFirstChild("HumanoidRootPart") then 
            colorCorrection.TintColor = Color3.fromRGB(255, 255, 255)
            return 
        end
        local rootPos = char.HumanoidRootPart.Position
        local maxIntensity = 0
        for grenade, radius in pairs(activeGrenades) do
            if grenade and grenade.Parent then
                local distance = (grenade.Position - rootPos).Magnitude
                local effectRange = radius * 2
                if distance < effectRange then
                    local intensity = 1 - (distance / effectRange)
                    if intensity > maxIntensity then maxIntensity = intensity end
                end
            end
        end
        colorCorrection.TintColor = Color3.fromRGB(255, 255 - (maxIntensity * 160), 255 - (maxIntensity * 160))
    end)
    table.insert(connections, renderConn)

        if obj.Name == "utility5Proxy" and obj:IsA("BasePart") then
            task.spawn(createSafetyWall, obj)
        end
    end
    local connDesc = workspace.DescendantAdded:Connect(onDescendantAdded)
    table.insert(connections, connDesc)
    
    task.spawn(function()
        for _, obj in pairs(workspace:GetDescendants()) do
            if obj.Name == "utility5Proxy" and obj:IsA("BasePart") then
                task.spawn(createSafetyWall, obj)
            end
        end
    end)
    AntiConnections["Grenade"] = connections
end

local function GetProfileFilename(profileName)
    profileName = profileName or "default";
    profileName = string.gsub(profileName, "[^%w%s_-]", "")
    if profileName == "" then profileName = "default" end
    return "acrocan0_hub_settings_" .. profileName .. ".json"
end

local function SaveSettings(profileName)
    local fileName = GetProfileFilename(profileName)
    if writefile then
        local success, err = pcall(function() writefile(fileName, HttpService:JSONEncode(_G_settings)) end)
        if success then ShowModeMsg("Profile Saved: " .. profileName);
        AddToRegistry(profileName); if RefreshProfilesUI then RefreshProfilesUI() end else ShowModeMsg("Error Saving Profile!") end

local function LoadSettings(profileName, silent)
    if profileName == "acrocan0 aim" then
        for k, v in pairs(Acrocan0_Hardcoded_Settings) do if _G_settings[k] ~= nil then _G_settings[k] = v end end
        local fileName = GetProfileFilename(profileName)
        if readfile and isfile and isfile(fileName) then
            pcall(function()
                local decoded = HttpService:JSONDecode(readfile(fileName))
                for k, v in pairs(decoded) do if _G_settings[k] ~= nil then _G_settings[k] = v end end
            end)
        end
        if not silent then ShowModeMsg("Loaded Official Profile: acrocan0 aim") end
        return true
    end
    local fileName = GetProfileFilename(profileName)
    if readfile and isfile and isfile(fileName) then
        local success, decoded = pcall(function() return HttpService:JSONDecode(readfile(fileName)) end)
        if success and type(decoded) == "table" then
            for k, v in pairs(decoded) do if _G_settings[k] ~= nil then _G_settings[k] = v end end
            if not silent then ShowModeMsg("Profile Loaded: " .. profileName) end
            return true
        end
    end
    if not silent then ShowModeMsg("Profile not found: " .. profileName) end
    return false
end

local function FadeUI(element, show)
    local targetAlpha = show and 0 or 1
    local ti = TweenInfo.new(0.3, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
    if show then element.Visible = true end
    if element:IsA("Frame") or element:IsA("TextButton") then TweenService:Create(element, ti, {BackgroundTransparency = targetAlpha}):Play() end
    if element:IsA("TextButton") then TweenService:Create(element, ti, {TextTransparency = targetAlpha}):Play() end
    for _, v in pairs(element:GetDescendants()) do
        if v:IsA("UIStroke") then TweenService:Create(v, ti, {Transparency = targetAlpha}):Play()
        elseif v:IsA("ImageLabel") then TweenService:Create(v, ti, {ImageTransparency = show and 0.6 or 1}):Play()
        elseif v:IsA("TextLabel") then TweenService:Create(v, ti, {TextTransparency = targetAlpha}):Play()
        elseif v:IsA("TextBox") then TweenService:Create(v, ti, {BackgroundTransparency = targetAlpha, TextTransparency = targetAlpha}):Play() end
    end
    if not show then task.delay(0.3, function() if not (_G_settings.UsePrediction and _G_settings.AutoPred) then element.Visible = false end end) end
end

local function CreateJoySlider(name, yPos, min, max, startVal, callback)
    local label = Instance.new("TextLabel", JoyEditorMain)
    label.Size = UDim2.new(1, -20, 0, 20)
    label.Position = UDim2.new(0, 10, 0, yPos)
    label.BackgroundTransparency = 1
    label.Text = name .. ": " .. math.floor(startVal)
    label.TextColor3 = Color3.new(1, 1, 1)
    label.Font = "GothamBold"
    label.TextSize = 12
    label.TextXAlignment = Enum.TextXAlignment.Left

    local container = Instance.new("Frame", JoyEditorMain)
    container.Size = UDim2.new(1, -20, 0, 20)
    container.Position = UDim2.new(0, 10, 0, yPos + 20)
    container.BackgroundColor3 = Color3.fromRGB(25, 25, 25)
    Instance.new("UICorner", container)
   
     local stroke = Instance.new("UIStroke", container)
    stroke.Color = Color3.fromRGB(100, 100, 100)

    local fill = Instance.new("Frame", container)
    local perc = (startVal - min) / (max - min)
    fill.Size = UDim2.new(perc, 0, 1, 0)
    fill.BackgroundColor3 = ScaryEdgeColor
    Instance.new("UICorner", fill)

    local isDragging = false
    container.InputBegan:Connect(function(input)
        if not ActiveJoySlider and (input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch) then 
            isDragging = true 
            ActiveJoySlider = true
        end
    end)
    fill.InputBegan:Connect(function(input)
        if not ActiveJoySlider and (input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch) then 
            isDragging = true 
            ActiveJoySlider = true
        end
    end)
 
    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then 
            if isDragging then
                isDragging = false 
                ActiveJoySlider = false
            end
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if isDragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) and JoyEditorMain.Visible then
            local relativeX = input.Position.X - container.AbsolutePosition.X
            local clampedX = math.clamp(relativeX, 0, container.AbsoluteSize.X)
            local p = clampedX / container.AbsoluteSize.X
            fill.Size = UDim2.new(p, 0, 1, 0)
            local val = min + (max - min) * p
            label.Text = name .. ": " .. math.floor(val)
            callback(val)
        end
    end)

    return function(newVal)
        local p = (newVal - min) / (max - min)
        fill.Size = UDim2.new(p, 0, 1, 0)
        label.Text = name .. ": " .. math.floor(newVal)
    end
end

local ESP_Drawings = {}

local function ClearESP(model)
    if ESP_Drawings[model] then
        ESP_Drawings[model].Box:Remove()
        ESP_Drawings[model].Tracer:Remove()
        ESP_Drawings[model].Arrow:Remove()
        ESP_Drawings[model] = nil
    end
end

RunService.RenderStepped:Connect(function()
    local cam = workspace.CurrentCamera
    local myChar = lp.Character
    local myRoot = myChar and myChar:FindFirstChild("HumanoidRootPart")

    -- مسح الـ ESP القديم
    for model, esp in pairs(ESP_Drawings) do
        if not model or not model.Parent or not LiveTargets[model] then 
            ClearESP(model) 
        end
    end

    for char, ent in pairs(LiveTargets) do
        if not char or not char:IsDescendantOf(workspace) then ClearESP(char); continue end
        if ent.isBot and not _G_settings.ESP_OnBots then ClearESP(char); continue end
        if not ent.isBot and _G_settings.IgnoredPlayers[ent.name] then ClearESP(char); continue end

        if not ESP_Drawings[char] and typeof(Drawing) == "table" then
            pcall(function()
                ESP_Drawings[char] = {Box = Drawing.new("Square"), Tracer = Drawing.new("Line"), Arrow = Drawing.new("Triangle")}
                ESP_Drawings[char].Box.Thickness = 1.5
                ESP_Drawings[char].Box.Filled = false
                ESP_Drawings[char].Tracer.Thickness = 1.5
                ESP_Drawings[char].Arrow.Thickness = 1
                ESP_Drawings[char].Arrow.Filled = true
            end)
        end

        local esp = ESP_Drawings[char]
        if not esp then continue end
        
        local targetColor = Color3.new(1, 0, 0)
        if not isMainGame and _G_settings.CheckTheBot and ent.isBot then targetColor = Color3.new(0, 1, 0) end
        esp.Box.Color = targetColor
        esp.Tracer.Color = targetColor;
        esp.Arrow.Color = targetColor

        local isAlive = char:FindFirstChild("Humanoid") and char.Humanoid.Health > 0

        if _G_settings.ESP_Enabled and isAlive and myRoot then
            local hrp = char:FindFirstChild("HumanoidRootPart")
            if hrp then
                local screenPos, onScreen = cam:WorldToViewportPoint(hrp.Position)
                local dist = (myRoot.Position - hrp.Position).Magnitude

                if onScreen then
                    if _G_settings.ESP_Boxes then
                        local top, _ = cam:WorldToViewportPoint((hrp.CFrame * CFrame.new(0, 3, 0)).Position)
                        local bottom, _ = cam:WorldToViewportPoint((hrp.CFrame * CFrame.new(0, -3, 0)).Position)
                        local height = math.abs(top.Y - bottom.Y)
                        local width = height * 0.6
                        esp.Box.Size = Vector2.new(width, height)
                        esp.Box.Position = Vector2.new(screenPos.X - width / 2, screenPos.Y - height / 2)
                        esp.Box.Visible = true
                    else esp.Box.Visible = false end

                    if _G_settings.ESP_Tracers then
                        esp.Tracer.From = Vector2.new(cam.ViewportSize.X / 2, cam.ViewportSize.Y)
                        esp.Tracer.To = Vector2.new(screenPos.X, screenPos.Y)
                        esp.Tracer.Visible = true
                    else esp.Tracer.Visible = false end
                else
                    esp.Box.Visible = false
                    esp.Tracer.Visible = false
                end

                if _G_settings.ESP_Arrows then
                    local alpha = 1 - math.clamp((dist - 15) / 105, 0, 1) 
                    if alpha > 0 then
                        local center = cam.ViewportSize / 2
                        local objSpace = cam.CFrame:PointToObjectSpace(hrp.Position)
                        local dir = Vector2.new(objSpace.X, -objSpace.Y).Unit
                        local radius = 150 
                        local arrowPos = center + dir * radius
                        local point1 = arrowPos + dir * 15
                        local point2 = arrowPos - dir * 10 + Vector2.new(-dir.Y, dir.X) * 10
                        local point3 = arrowPos - dir * 10 - Vector2.new(-dir.Y, dir.X) * 10
                        esp.Arrow.PointA = point1
                        esp.Arrow.PointB = point2;
                        esp.Arrow.PointC = point3
                        esp.Arrow.Transparency = alpha
                        esp.Arrow.Visible = true
                    else esp.Arrow.Visible = false end
                else esp.Arrow.Visible = false end
            else esp.Box.Visible = false
                esp.Tracer.Visible = false;
                esp.Arrow.Visible = false end
        else esp.Box.Visible = false
            esp.Tracer.Visible = false;
            esp.Arrow.Visible = false end
    end
end)

local LastFPSUpdate = tick()
local FrameCount = 0

RunService.RenderStepped:Connect(function(deltaTime)
    FrameCount = FrameCount + 1
    if tick() - LastFPSUpdate >= 0.5 then
        if _G_settings.ShowFPSCounter then FPSDisplay.Text = "FPS: " .. math.floor(FrameCount / (tick() - LastFPSUpdate)) end
        FrameCount = 0; LastFPSUpdate = tick()
    end
    
    local cam = workspace.CurrentCamera
    
    if FOVCircle then
        FOVCircle.Position = Vector2.new(cam.ViewportSize.X/2, (cam.ViewportSize.Y/2) + _G_settings.CrosshairOffsetY)
        FOVCircle.Color = Color3.fromRGB(255, 0, 0)
        FOVCircle.Radius = _G_settings.FOV 
    end
    
    if _G_settings.Enabled then
        if _G_settings.RightClickAim and not isAimKeyDown then
            -- لا تصوب إذا كان خيار الكليك اليمين مفعل والزر غير مضغوط
        else
            local myChar = lp.Character
            local myRoot = myChar and myChar:FindFirstChild("HumanoidRootPart")
            if not myRoot then return end
          
            -- OPTIMIZED AIMBOT LOGIC
            local bestTarget = nil
            local bestPart = nil
            local bestVal = math.huge
            
            for v, ent in pairs(LiveTargets) do
                if ent.isBot and not _G_settings.AimAtBots then continue end
                if not ent.isBot and _G_settings.IgnoredPlayers[ent.name] then continue end

                if v and v:FindFirstChild("Humanoid") and v.Humanoid.Health > 0 then
                    local targetPart = v:FindFirstChild(_G_settings.TargetPart)
                    if targetPart then
                        local screenPos, onScreen = cam:WorldToViewportPoint(targetPart.Position)
                        if onScreen then
                            local mouseDist = (Vector2.new(screenPos.X, screenPos.Y) - Vector2.new(cam.ViewportSize.X/2, (cam.ViewportSize.Y/2) + _G_settings.CrosshairOffsetY)).Magnitude
                            if mouseDist <= _G_settings.FOV then
                                local worldDist = (myRoot.Position - targetPart.Position).Magnitude
                                if worldDist <= _G_settings.MaxDist then
                                    -- نختار الأفضل بناءً على الأولوية (بدون ما نستخدم table.sort)
                                    local checkVal = _G_settings.LowHealthFocus and v.Humanoid.Health or mouseDist
                                    
                                    if checkVal < bestVal then
                                        local isVisible = true
                                        if _G_settings.WallCheck then
                                            local rayParams = RaycastParams.new()
                                            rayParams.FilterType = Enum.RaycastFilterType.Exclude
                                            rayParams.FilterDescendantsInstances = {myChar, v}
                                            local ray = workspace:Raycast(cam.CFrame.Position, (targetPart.Position - cam.CFrame.Position).Unit * worldDist, rayParams)
                                            if ray then isVisible = false end
                                        end
                                        
                                        if isVisible then
                                            bestVal = checkVal
                                            bestTarget = v
                                            bestPart = targetPart
                                        end
                                    end
                                end
                            end
                        end
                    end
                end
            end
            
            if bestTarget and bestPart then
                local aimPos = bestPart.Position
 
                if _G_settings.UsePrediction then
                    local currentPrediction = _G_settings.Prediction
                    if _G_settings.AutoPred then
                        local dist = (myRoot.Position - bestPart.Position).Magnitude
                        currentPrediction = _G_settings.AutoPredBase
                        if dist <= _G_settings.DistThreshold then currentPrediction = currentPrediction + (dist * _G_settings.CloseMult)
                        else currentPrediction = currentPrediction + (_G_settings.DistThreshold * _G_settings.CloseMult) + ((dist - _G_settings.DistThreshold) * _G_settings.FarMult) end
                        if UI_Inputs.Pred then UI_Inputs.Pred.Text = string.format("%.3f", currentPrediction) end
                    end
                    local vel = bestPart.AssemblyLinearVelocity
                    if vel.Magnitude > 130 then vel = vel.Unit * 130 end
                    local predX = vel.X * currentPrediction
                    local predY = vel.Y * currentPrediction * _G_settings.VerticalStrength;
                    local predZ = vel.Z * currentPrediction
                    aimPos = aimPos + Vector3.new(predX, predY, predZ)
                end
                
                local targetLook = CFrame.lookAt(cam.CFrame.Position, aimPos)
                local offsetAngle = math.rad((_G_settings.CrosshairOffsetY / cam.ViewportSize.Y) * cam.FieldOfView)
                targetLook = targetLook * CFrame.Angles(offsetAngle, 0, 0)
                
                if targetLook.X == targetLook.X and targetLook.Y == targetLook.Y and targetLook.Z == targetLook.Z then
                    local smoothVal = 1
                    if _G_settings.Smoothness and _G_settings.Smoothness > 0 then smoothVal = math.clamp(1 / (_G_settings.Smoothness * deltaTime * 60), 0.01, 1) end
                    cam.CFrame = cam.CFrame:Lerp(targetLook, smoothVal)
                end
             end
        end
    end
end)

task.spawn(StartSequence)
