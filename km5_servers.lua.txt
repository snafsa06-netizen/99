local function GetDynamicScale()
    local viewport = Camera.ViewportSize
    local ratio = 1
    if viewport.X > 0 and viewport.Y > 0 then
        ratio = math.min(viewport.X / 1280, viewport.Y / 720)
    end
    local scaleVal = 0.5 + (SizeIndex * 0.1) 
    return ratio * scaleVal
end

local function ApplyColorTheme(glowColor, activeBtn)
    TargetGlowColor = glowColor
    local r, g, b = glowColor.R, glowColor.G, glowColor.B
    local mainColor = Color3.new(r * 0.14, g * 0.14, b * 0.14)
    local boxColor  = Color3.new(r * 0.08, g * 0.08, b * 0.08)

    CurrentBoxBgColor  = boxColor

        if box and box:IsA("Frame") and (box.Name == "MainFrame" or box.Name == "EditorMain") then

local ESP_Cache = {}

local function ClearESP(model)
    if ESP_Cache[model] then
        if ESP_Cache[model].Highlight then ESP_Cache[model].Highlight:Destroy() end
        if ESP_Cache[model].Tracer then ESP_Cache[model].Tracer:Remove() end
        ESP_Cache[model] = nil
    end
end

local function GetKaijus()
    local targets = {}
    for _, p in pairs(game.Players:GetPlayers()) do
        if p ~= lp and p.Character then table.insert(targets, p.Character) end
    end
    local map = workspace:FindFirstChild("Map")
    local folders = {workspace}
    if map and map:FindFirstChild("CityFolder") then table.insert(folders, map.CityFolder) end
    if workspace:FindFirstChild("Buildings") then table.insert(folders, workspace.Buildings) end
 
    for _, folder in pairs(folders) do
         for _, model in pairs(folder:GetChildren()) do
            if model:IsA("Model") and model ~= lp.Character then
                local name = string.lower(model.Name)
                if string.find(name, "kaiju") or model:FindFirstChild("Humanoid") then
                    table.insert(targets, model)
                end
            end
        end
    end
    return targets
end

RunService.RenderStepped:Connect(function()
    local camera = workspace.CurrentCamera
    if not _G.ESP_Enabled then
        for model, _ in pairs(ESP_Cache) do ClearESP(model) end
        return
    end

    local currentKaijus = GetKaijus()
    local validModels = {}

    for _, model in ipairs(currentKaijus) do
        local root = model:FindFirstChild("HumanoidRootPart")
        local hum = model:FindFirstChild("Humanoid")
        
        if root and hum and hum.Health > 0 then
            validModels[model] = true
             if not ESP_Cache[model] then
                 ESP_Cache[model] = {}
                 local hl = Instance.new("Highlight")
                hl.Name = "KaijuESP"
                hl.FillColor = Color3.fromRGB(255, 0, 0)
                hl.FillTransparency = 0.35
                hl.OutlineColor = Color3.fromRGB(255, 0, 0)
                hl.OutlineTransparency = 0
                hl.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
                hl.Adornee = model
                pcall(function() hl.Parent = CoreGui end)
                if not hl.Parent then hl.Parent = PlayerGui end
                 ESP_Cache[model].Highlight = hl

                pcall(function()
                    local tracer = Drawing.new("Line")
                    tracer.Visible = false
                    tracer.Color = Color3.fromRGB(255, 255, 255)
                    tracer.Thickness = 1.5
                    tracer.Transparency = 1
                    ESP_Cache[model].Tracer = tracer
                end)
            end

            if ESP_Cache[model].Tracer then
                if _G.ESP_Tracers then
                    local vector, onScreen = camera:WorldToViewportPoint(root.Position)
                    if onScreen then
                        ESP_Cache[model].Tracer.From = Vector2.new(camera.ViewportSize.X / 2, camera.ViewportSize.Y)
                        ESP_Cache[model].Tracer.To = Vector2.new(vector.X, vector.Y)
                        ESP_Cache[model].Tracer.Visible = true
                    else
                        ESP_Cache[model].Tracer.Visible = false
                    end
                else
                    ESP_Cache[model].Tracer.Visible = false
                end
            end
         else
            ClearESP(model)
        end
    end

    for model, _ in pairs(ESP_Cache) do
        if not validModels[model] then
              ClearESP(model)
        end
    end
end)

local isSequenceRunning = false

local function findPlayButton()
    for _, v in pairs(pgui:GetDescendants()) do
        if v:IsA("TextButton") or v:IsA("ImageButton") then
            if string.lower(v.Name) == "play" then return v end
        end
    end
    return nil
end

local function findSpawnButton()
    for _, v in pairs(pgui:GetDescendants()) do
        if v:IsA("TextButton") or v:IsA("ImageButton") then
            if v.Parent and v.Parent.Name == "Spawn_ScreenGui" then return v end
            if string.lower(v.Name) == "spawn" then return v end
        end
    end
    return nil
end

local function findSelectButton()
    for _, v in pairs(pgui:GetDescendants()) do
        if v:IsA("TextButton") or v:IsA("ImageButton") then
            if string.lower(v.Name) == "select" then return v end
        end
    end
    return nil
end

local function pressButton(btn, extraXOffset)
    if not _G_AutoClicker.Enabled then return end
    extraXOffset = extraXOffset or 0 
    
    if btn and btn.AbsoluteSize.X > 0 and btn.Visible then
        local visualX = btn.AbsolutePosition.X + (btn.AbsoluteSize.X / 2)
        local visualY = btn.AbsolutePosition.Y + (btn.AbsoluteSize.Y / 2)
        local parentGui = btn:FindFirstAncestorOfClass("ScreenGui")
        
        if parentGui and not parentGui.IgnoreGuiInset then
            local inset = GuiService:GetGuiInset()
            visualX = visualX + inset.X
            visualY = visualY + inset.Y
        end

        if visualX <= 10 or visualY <= 10 then return end
         
        pcall(function()
            local hardwareClickX = visualX + _G_AutoClicker.X_Offset + extraXOffset
            local hardwareClickY = visualY + _G_AutoClicker.Y_Offset
            vim:SendMouseButtonEvent(hardwareClickX, hardwareClickY, 0, true, game, 1)
            task.wait(0.03)
            vim:SendMouseButtonEvent(hardwareClickX, hardwareClickY, 0, false, game, 1)
            vim:SendTouchEvent({{Position = Vector2.new(hardwareClickX, hardwareClickY), ID = 0, State = Enum.UserInputState.Begin}})
            task.wait(0.03)
            vim:SendTouchEvent({{Position = Vector2.new(hardwareClickX, hardwareClickY), ID = 0, State = Enum.UserInputState.End}})
        end)
    end
end

local function runSpawnSequence()
    if isSequenceRunning or not _G_AutoClicker.Enabled then return end
    isSequenceRunning = true
    
    task.spawn(function()
        task.wait(4)
        if not _G_AutoClicker.Enabled then isSequenceRunning = false return end
        local playBtn = findPlayButton()
        if playBtn then pressButton(playBtn) end
        
        task.wait(2)
        if not _G_AutoClicker.Enabled then isSequenceRunning = false return end

        local selectBtn = findSelectButton()
        if selectBtn then
            pressButton(selectBtn)
        else
            task.wait(1)
            selectBtn = findSelectButton()
            if selectBtn then pressButton(selectBtn) end
        end

        task.wait(2)
        if not _G_AutoClicker.Enabled then isSequenceRunning = false return end

        local spawnBtn = findSpawnButton()
        if spawnBtn then
            pressButton(spawnBtn, _G_AutoClicker.Spawn_Shift) 
        else
            task.wait(1)
            spawnBtn = findSpawnButton()
            if spawnBtn then pressButton(spawnBtn, _G_AutoClicker.Spawn_Shift) end
        end
        
        task.wait(20)
        
        if _G_AutoClicker.Enabled then
            if lp.Character and lp.Character:FindFirstChild("Humanoid") then
                if lp.Character.Humanoid.Health <= 0 then
                    playBtn = findPlayButton()
                    if playBtn then pressButton(playBtn) end
                    task.wait(2)
                    selectBtn = findSelectButton()
                    if selectBtn then pressButton(selectBtn) end
                    task.wait(2)
                    spawnBtn = findSpawnButton()
                    if spawnBtn then pressButton(spawnBtn, _G_AutoClicker.Spawn_Shift) end
                end
            else
                playBtn = findPlayButton()
                if playBtn then pressButton(playBtn) end
                task.wait(2)
                selectBtn = findSelectButton()
                if selectBtn then pressButton(selectBtn) end
                task.wait(2)
                spawnBtn = findSpawnButton()
                if spawnBtn then pressButton(spawnBtn, _G_AutoClicker.Spawn_Shift) end
            end
        end
        isSequenceRunning = false
    end)
end

local function monitorHealth(character)
    if not character then return end
    local humanoid = character:WaitForChild("Humanoid", 10)
    if humanoid then
        if humanoid.Health <= 0 then runSpawnSequence() end
        humanoid.HealthChanged:Connect(function(currentHealth)
            if currentHealth <= 0 then runSpawnSequence() end
        end)
        humanoid.Died:Connect(function() runSpawnSequence() end)
    end
end

task.spawn(function()
    if lp.Character then monitorHealth(lp.Character) end
    lp.CharacterAdded:Connect(function(newCharacter)
        monitorHealth(newCharacter)
    end)
    while true do
        task.wait(5)
        if _G_AutoClicker.Enabled and not isSequenceRunning then
            local char = lp.Character
            if not char or not char:FindFirstChild("HumanoidRootPart") or (char:FindFirstChild("Humanoid") and char.Humanoid.Health <= 0) then
                runSpawnSequence()
            end
        end
    end
end)

task.spawn(function()
    while true do
        task.wait(30)
        if _G_AutoClicker.Enabled and not isSequenceRunning then
             runSpawnSequence()
         end
    end
end)

local function ApplyResponsiveScale(guiElement)
    local uiScale = Instance.new("UIScale", guiElement)
    local function UpdateScale()
        if Camera then
            local viewport = Camera.ViewportSize
            local baseRes = Vector2.new(1280, 720) 
            local scale = math.min(viewport.X / baseRes.X, viewport.Y / baseRes.Y)
            uiScale.Scale = math.clamp(scale * 1.3, 1.0, 6.25) 
        end
    end
    Camera:GetPropertyChangedSignal("ViewportSize"):Connect(UpdateScale)
    UpdateScale()
end

local function SafeTextScale(obj, optimalSize)
    if obj:IsA("TextLabel") or obj:IsA("TextButton") or obj:IsA("TextBox") then
        local maxScaleFactor = 2.0
        local targetSize = math.floor(optimalSize * maxScaleFactor)
        
        obj.TextScaled = true
        obj.RichText = true
        
        local constraint = obj:FindFirstChildOfClass("UITextSizeConstraint")
        if not constraint then
            constraint = Instance.new("UITextSizeConstraint")
            constraint.Parent = obj
        end
        constraint.MaxTextSize = targetSize
        constraint.MinTextSize = 6 
    end
end

local function AnimateButton(btn, clickCallback, isColorSquare)
    local uiScale = btn:FindFirstChildOfClass("UIScale") or Instance.new("UIScale", btn)
    local tweenInfoSlow = TweenInfo.new(0.22, Enum.EasingStyle.Sine, Enum.EasingDirection.Out)
    local tweenInfoPress = TweenInfo.new(0.1, Enum.EasingStyle.Sine, Enum.EasingDirection.Out)

    local activeScaleTween = nil
    local activeBtnTween = nil

    local function playTween(targetScale, targetBgTrans, targetTextTrans, speedInfo)
        if activeScaleTween then activeScaleTween:Cancel() end
        if activeBtnTween then activeBtnTween:Cancel() end

        activeScaleTween = TweenService:Create(uiScale, speedInfo, {Scale = targetScale})
        activeBtnTween = TweenService:Create(btn, speedInfo, {
            BackgroundTransparency = targetBgTrans,
            TextTransparency = targetTextTrans
        })

        activeScaleTween:Play()
        activeBtnTween:Play()
    end

    local isImage = btn:IsA("ImageButton")
    local baseBg = isImage and 1 or 0.90   
    local hoverBg = isImage and 1 or 0.96 
    local pressBg = isImage and 1 or 1.0  
    
    if not isImage then
        btn.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
        btn.BackgroundTransparency = baseBg
    end

    btn.MouseEnter:Connect(function() 
        local isActive = btn:GetAttribute("IsActive")
        local targetScale = isActive and 1.10 or 1.04
        local targetBg = isActive and 1 or hoverBg 
        playTween(targetScale, targetBg, 0, tweenInfoSlow) 
    end)
    
    btn.MouseLeave:Connect(function() 
        local isActive = btn:GetAttribute("IsActive")
        local targetScale = isActive and 1.10 or 1.0
        local targetBg = isActive and 1 or baseBg 
        playTween(targetScale, targetBg, 0, tweenInfoSlow) 
    end)
    
    btn.MouseButton1Down:Connect(function() 
        local isActive = btn:GetAttribute("IsActive")
        local targetScale = isActive and 1.05 or 0.94
        playTween(targetScale, pressBg, 0.1, tweenInfoPress) 
    end)
    
    btn.MouseButton1Up:Connect(function() 
        local isActive = btn:GetAttribute("IsActive")
        local targetScale = isActive and 1.10 or 1.04
        playTween(targetScale, hoverBg, 0, tweenInfoSlow) 
    end)

    btn.MouseButton1Click:Connect(function()
        ClickSound:Play()
        if clickCallback then clickCallback() end
    end)
end

local function ApplyJoystickSettings()
    if GrayGui and GrayGui:FindFirstChild("TouchBlocker") then
         local main = GrayGui.TouchBlocker.MainFrame
         main.Size = UDim2.new(0, _G_settings.JoySize, 0, _G_settings.JoySize)
         main.Position = UDim2.new(0, _G_settings.JoyPosX, 0.5, _G_settings.JoyPosY)
         local stick = main.Stick
        local half = _G_settings.JoySize / 2
         stick.Size = UDim2.new(0, half, 0, half)
         if not activeTouch then
            stick.Position = UDim2.new(0.5, -half/2, 0.5, -half/2)
         end
     end
end

local function stopMoving()
    activeTouch = nil
    moveDir = Vector3.new(0,0,0)
    if GrayGui and GrayGui:FindFirstChild("TouchBlocker") then
        local stick = GrayGui.TouchBlocker.MainFrame.Stick
        local half = _G_settings.JoySize / 2
        stick:TweenPosition(UDim2.new(0.5, -half/2, 0.5, -half/2), "Out", "Back", 0.1)
    end
    SetMoveVector(Vector3.new(0, 0, 0))
end

local function setupGrayJoy()
    local sg = Instance.new("ScreenGui", pgui)
    sg.Name = "Gray_Movement"
    sg.DisplayOrder = -100
    sg.Enabled = false
    sg.ResetOnSpawn = false 
    sg.IgnoreGuiInset = true 
    sg.ScreenInsets = Enum.ScreenInsets.None
     
    local touchBlocker = Instance.new("Frame", sg)
     touchBlocker.Name = "TouchBlocker"
     touchBlocker.Size = UDim2.new(0, 400, 0, 180)
    touchBlocker.Position = UDim2.new(0, 10, 1, -200)
    touchBlocker.BackgroundTransparency = 1
    touchBlocker.ZIndex = -10
    touchBlocker.Active = true 
    
    local main = Instance.new("Frame", touchBlocker)
    main.Name = "MainFrame"
    main.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
    main.BackgroundTransparency = 0.80
    main.ZIndex = -9
    main.Active = true 
     Instance.new("UICorner", main).CornerRadius = UDim.new(1, 0)
    ApplyResponsiveScale(touchBlocker) 
  
    local stick = Instance.new("Frame", main)
    stick.Name = "Stick"
    stick.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
    stick.BackgroundTransparency = 0.80
    stick.ZIndex = -8
    Instance.new("UICorner", stick).CornerRadius = UDim.new(1, 0)

    GrayGui = sg
    ApplyJoystickSettings()

    main.InputBegan:Connect(function(input)
         if (input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1) then
            activeTouch = input
            local center = main.AbsolutePosition + (main.AbsoluteSize / 2)
            local diff = Vector2.new(input.Position.X - center.X, input.Position.Y - center.Y)
            local limit = main.AbsoluteSize.X / 2
            local dist = math.min(diff.Magnitude, limit)
            if dist > 0.05 then
                  local dir = diff.Unit
                 SetMoveVector(Vector3.new(dir.X * (dist/limit), 0, dir.Y * (dist/limit)))
             end
         end
    end)
   
    UserInputService.InputChanged:Connect(function(input)
        if input == activeTouch and sg.Enabled then
             local center = main.AbsolutePosition + (main.AbsoluteSize / 2)
             local diff = Vector2.new(input.Position.X - center.X, input.Position.Y - center.Y)
            local limit = main.AbsoluteSize.X / 2
            local dist = math.min(diff.Magnitude, limit)
            local dir = diff.Unit
            local half = _G_settings.JoySize / 2
            stick.Position = UDim2.new(0.5, (dir.X * dist) - half/2, 0.5, (dir.Y * dist) - half/2)
   
             if dist > 0.05 then 
                 SetMoveVector(Vector3.new(dir.X * (dist/limit), 0, dir.Y * (dist/limit))) 
            else 
                 SetMoveVector(Vector3.new(0, 0, 0)) 
            end
        end
    end)
    UserInputService.InputEnded:Connect(function(input) if input == activeTouch then stopMoving() end end)
     UserInputService.WindowFocusReleased:Connect(stopMoving)
   
    return sg
end

local function UpdateGrayMovementState()
    local shouldBeActive  = _G_settings.CustomJoystick
    local touchGui = pgui:FindFirstChild("TouchGui")
    local function toggleDefaultThumbstick(visible)
        if touchGui then
             local frame = touchGui:FindFirstChild("TouchControlFrame")
            if frame then
                  for _, v in pairs(frame:GetChildren()) do
                     if string.find(v.Name, "Thumbstick") then
                        for _, desc in pairs(v:GetDescendants()) do if desc:IsA("ImageLabel") or desc:IsA("ImageButton") then desc.ImageTransparency = visible and 0 or 1 end end
                         v.BackgroundTransparency = 1
                     end
                   end
            end
         end
    end
    if shouldBeActive then
        toggleDefaultThumbstick(false)
        if not GrayGui then GrayGui = setupGrayJoy() end
         GrayGui.Enabled = true
         ApplyJoystickSettings()
    else
        toggleDefaultThumbstick(true)
        if GrayGui and GrayGui.Enabled then GrayGui.Enabled = false; stopMoving() end
    end
end

local TotalLayers = 150 
local LayersData = {}
local ZoomRatio = 2 

for i = 1, TotalLayers do
    local Layer = Instance.new("Frame")
    Layer.Parent = GlowContainer
    Layer.AnchorPoint = Vector2.new(0.5, 0.5)
    Layer.BorderSizePixel = 0
    Layer.ZIndex = 4 
   
    local baseSize = (40 + (i * 2.5)) * ZoomRatio 
    Layer.Size = UDim2.new(0, baseSize, 0, baseSize)
    local progress = (TotalLayers - i) / TotalLayers
    Layer.BackgroundTransparency = 1 
    Instance.new("UICorner", Layer).CornerRadius = UDim.new(1, 0)
    
    table.insert(LayersData, { Instance = Layer, BaseSize = baseSize, OffsetFactor = (i / TotalLayers), Progress = progress })
end

task.spawn(function()
    local timeCounter = 0
    while true do
        timeCounter = timeCounter + 0.007 
        CurrentGlowColor = CurrentGlowColor:Lerp(TargetGlowColor, 0.03)
   
        for _, data in pairs(LayersData) do
            if data.Instance and data.Instance.Parent then
                local offsetX, offsetY, sizePulse
                if CurrentAnimStyle == "1v" then
                    local waveAngle = timeCounter * 0.6 + (data.OffsetFactor * math.pi * 1.5)
                    offsetX = math.sin(waveAngle) * 120 
                    offsetY = math.cos(waveAngle * 0.5) * 40 
                    sizePulse = 1 + (math.sin(waveAngle * 0.8) * 0.12)
                elseif CurrentAnimStyle == "2v" then 
                    local waveAngle = timeCounter * 1.2 + (data.OffsetFactor * math.pi * 2.5)
                    offsetX = math.sin(waveAngle) * 80
                    offsetY = math.cos(waveAngle * 0.8) * 80
                    sizePulse = 1 + (math.sin(timeCounter * 1.5 + data.OffsetFactor) * 0.25)
                elseif CurrentAnimStyle == "3v" then
                    local speed = 0.15 + (data.OffsetFactor * 0.35)
                      local seed = data.OffsetFactor * 100
                    local layerAngle = timeCounter * (speed * 1) + seed
                    offsetX = (math.sin(layerAngle) * 40) - 50 
                    offsetY = math.cos(layerAngle * 0.8) * 20
                    sizePulse = 1 + (math.sin(layerAngle * 1.2) * 0.25)
                end
                
                local distanceFactor = 0.8 * ZoomRatio * 1.5
                data.Instance.Position = UDim2.new(0.5, offsetX * distanceFactor, 0.5, offsetY * distanceFactor)
                
                local currentSize = data.BaseSize * sizePulse
                data.Instance.Size = UDim2.new(0, currentSize, 0, currentSize)
                
                local baseTrans = 0.983 + (data.Progress * 0.008) 
                local finalTrans = 1 - ((1 - baseTrans) * GlowAlpha)
                data.Instance.BackgroundTransparency = finalTrans
                data.Instance.BackgroundColor3 = CurrentGlowColor
            end
        end
        task.wait()
    end
end)

local function CreateSlider(parent, nameKey, min, max, startVal, callback)
    local container = Instance.new("Frame", parent)
    container.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
    container.BackgroundTransparency = 0.90
    Instance.new("UICorner", container).CornerRadius = UDim.new(0, 4)
    table.insert(AllUIBoxes, container)

    local label = Instance.new("TextLabel", container)
    label:SetAttribute("SliderKey", nameKey)
    label:SetAttribute("CurrentValue", startVal)
    
    local T = Translations[CurrentLang] or Translations.EN
    label.Size = UDim2.new(1, -10, 0, 25)
    label.Position = UDim2.new(0, 5, 0, 5)
    label.BackgroundTransparency = 1
    label.Text = (T[nameKey] or nameKey) .. ": " .. math.floor(startVal)
    label.TextColor3 = Color3.fromRGB(255, 255, 255)
    label.Font = Enum.Font.Gotham
    SafeTextScale(label, 8)
    label.TextXAlignment = Enum.TextXAlignment.Left

    local sliderBg = Instance.new("Frame", container)
    sliderBg.Size = UDim2.new(1, -20, 0, 8)
    sliderBg.Position = UDim2.new(0, 10, 0, 32)
    sliderBg.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
    sliderBg.BackgroundTransparency = 0.80
    sliderBg.BorderSizePixel = 0
    Instance.new("UICorner", sliderBg).CornerRadius = UDim.new(1, 0)

    local fill = Instance.new("Frame", sliderBg)
    local perc = (startVal - min) / (max - min)
    fill.Size = UDim2.new(perc, 0, 1, 0)
    fill.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
    fill.BackgroundTransparency = 0.80
    Instance.new("UICorner", fill).CornerRadius = UDim.new(1, 0)

    local isDragging = false
    
    local function updateSlider(input)
        local relativeX = input.Position.X - sliderBg.AbsolutePosition.X
        local clampedX = math.clamp(relativeX, 0, sliderBg.AbsoluteSize.X)
        local p = clampedX / sliderBg.AbsoluteSize.X
        fill.Size = UDim2.new(p, 0, 1, 0)
        local val = min + (max - min) * p
        label:SetAttribute("CurrentValue", val)
        local CurrentT = Translations[CurrentLang] or Translations.EN
         label.Text = (CurrentT[nameKey] or nameKey) .. ": " .. math.floor(val)
        callback(val)
    end

    sliderBg.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then 
            isDragging = true;
            updateSlider(input)
        end
    end)
    
    fill.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then 
            isDragging = true; updateSlider(input)
        end
    end)

    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then 
            isDragging = false 
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if isDragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
            updateSlider(input)
        end
    end)
end

local function ApplyNoEffects(enabled)

        Lighting.GlobalShadows = false
        for _, v in pairs(Lighting:GetChildren()) do if v:IsA("PostEffect") then v.Enabled = false end end

        EffectConnection = workspace.DescendantAdded:Connect(function(v) if _G_settings.NoEffects then task.wait();
cleanEffects(v) end end)
    else 
        if EffectConnection then EffectConnection:Disconnect() end 
    end
end

-- [[ Auto Server Switch (كل 30 ثانية) ]]
-- ==========================================
local _autoSwitchThread = nil
local _serverJoinTime = os.time()

local function _stopAutoSwitch()
    if _autoSwitchThread then task.cancel(_autoSwitchThread); _autoSwitchThread = nil end
    AutoSwitchDisplay.Visible = false
end

local function _startAutoSwitch()
    _stopAutoSwitch()
    if not AutoSwitch_Enabled then return end
    _serverJoinTime = os.time()
    AutoSwitchDisplay.Visible = true
    _autoSwitchThread = task.spawn(function()
        local SWITCH_AFTER = 30
        while AutoSwitch_Enabled do
            local elapsed = os.time() - _serverJoinTime
            local remaining = SWITCH_AFTER - elapsed
            if remaining <= 0 then
                AutoSwitchDisplay.Text = "🔀 Switching..."
                ShowModeMsg("Auto Switch: Finding server...")
                local ok, result = pcall(function()
                    return HttpService:JSONDecode(game:HttpGet(
                        "https://games.roblox.com/v1/games/" .. game.PlaceId .. "/servers/Public?sortOrder=Asc&limit=10"
                    ))
                end)
                if ok and result and result.data then
                    for _, srv in ipairs(result.data) do
                        if srv.id ~= game.JobId and srv.playing < srv.maxPlayers then
                            _doHop(srv.id)
                            return
                        end
                    end
                end
                ShowModeMsg("Auto Switch: No server found, retrying...")
                _serverJoinTime = os.time()
                task.wait(5)
            else
                AutoSwitchDisplay.Text = string.format("🔀 Switch in: %ds", remaining)
                task.wait(1)
            end
        end
        _stopAutoSwitch()
    end)
end

    else

    end

end)

local function SwitchTab(activeKey)
    if currentActiveTab == activeKey then return end
    currentActiveTab = activeKey

    OpenSound:Play()

    for key, data in pairs(TabData) do
        local tabBtn = UI_Elements[key]

        if key == activeKey then
            data.CG.Visible = true
            TweenService:Create(data.CG, TweenInfo.new(0.3, Enum.EasingStyle.Sine, Enum.EasingDirection.Out), {GroupTransparency = 0}):Play()

             if tabBtn then
                local uiScale = tabBtn:FindFirstChildOfClass("UIScale")
                 tabBtn:SetAttribute("IsActive", true)
                if uiScale then TweenService:Create(uiScale, TweenInfo.new(0.3, Enum.EasingStyle.Sine, Enum.EasingDirection.Out), {Scale = 1.10}):Play() end
                TweenService:Create(tabBtn, TweenInfo.new(0.3, Enum.EasingStyle.Sine, Enum.EasingDirection.Out), {BackgroundTransparency = 1}):Play()
            end
        else
            if data.CG.Visible then
                 local fadeTween = TweenService:Create(data.CG, TweenInfo.new(0.3, Enum.EasingStyle.Sine, Enum.EasingDirection.Out), {GroupTransparency = 1})
                fadeTween:Play()
    
                task.spawn(function()
                    fadeTween.Completed:Wait()
                    if currentActiveTab ~= key then
                            data.CG.Visible = false
                    end
                end)
            end

            if tabBtn then
                 local uiScale = tabBtn:FindFirstChildOfClass("UIScale")
                tabBtn:SetAttribute("IsActive", false)

                TweenService:Create(tabBtn, TweenInfo.new(0.3, Enum.EasingStyle.Sine, Enum.EasingDirection.Out), {BackgroundTransparency = 0.90}):Play()
            end
        end
    end
end
