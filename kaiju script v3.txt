RunService.Heartbeat:Connect(function(deltaTime)
    local char = lp.Character
    if not char then return end
    local hum = char:FindFirstChildOfClass("Humanoid")
    local root = char:FindFirstChild("HumanoidRootPart")
    
    if hum and root and hum.Health > 0 then
        
        -- No Stop Logic
        if NoStopEnabled then
            if hum.PlatformStand then hum.PlatformStand = false end
            if hum.AutoRotate == false then hum.AutoRotate = true end
            if root.Anchored then root.Anchored = false end
        end

        -- القوة الجبرية للسرعة (عبر CFrame وليس WalkSpeed)
        if isSpeedForced and forceSpeed > 16 then
            if hum.MoveDirection.Magnitude > 0 then
                local extraSpeed = forceSpeed - hum.WalkSpeed
                if extraSpeed > 0 then
                    root.CFrame = root.CFrame + (hum.MoveDirection * (extraSpeed * deltaTime))
                end
            end
        end
        
        -- القوة الجبرية للنط
        if isJumpForced then
            hum.UseJumpPower = true
            hum.JumpPower = forceJump
        end
    end
end)

local currentTarget = nil
local playerTarget = nil
local isWaitingForRespawn = false
local lastWReset = tick()
local smoothness = 0.15 

local function getHorizontalDistance(pos1, pos2)
    return (Vector2.new(pos1.X, pos1.Z) - Vector2.new(pos2.X, pos2.Z)).Magnitude
end

local function updateGlobalTargets()
    local char = lp.Character
    local root = char and char:FindFirstChild("HumanoidRootPart")
    if not root then return end
    
    local rootPos = root.Position
    local closestBuilding = nil
    local closestBuildingDist = math.huge 
    local closestPlayer = nil
    local closestPlayerDist = 70 

    for _, p in pairs(game.Players:GetPlayers()) do
        if p ~= lp and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
            local dist = getHorizontalDistance(rootPos, p.Character.HumanoidRootPart.Position)
            if dist < closestPlayerDist then
                closestPlayerDist = dist
                closestPlayer = p.Character
            end
        end
    end

    local buildings = workspace.Map:FindFirstChild("RenderBuildings")
    if buildings then
        for _, b in pairs(buildings:GetChildren()) do
            if not b:FindFirstChild("0") and b.Name ~= "Tree_Pine" then
                local dist = getHorizontalDistance(rootPos, b:GetPivot().Position)
                if dist < closestBuildingDist then
                    closestBuildingDist = dist
                    closestBuilding = b
                end
            end
        end
    end
    playerTarget = closestPlayer
    currentTarget = closestBuilding
end

RunService.RenderStepped:Connect(function()
    if not _G.AutoFarm then return end
    local camera = workspace.CurrentCamera
    local char = lp.Character
    local root = char and char:FindFirstChild("HumanoidRootPart")
    
    local targetPos = nil
    if playerTarget then targetPos = playerTarget:GetPivot().Position
    elseif currentTarget then targetPos = currentTarget:GetPivot().Position end

    if targetPos and camera then
        local targetCFrame = CFrame.lookAt(camera.CFrame.Position, targetPos)
        camera.CFrame = camera.CFrame:Lerp(targetCFrame, smoothness)
        
        if root then
            local bodyLook = CFrame.lookAt(root.Position, Vector3.new(targetPos.X, root.Position.Y, targetPos.Z))
            root.CFrame = root.CFrame:Lerp(bodyLook, smoothness)
        end
    end
end)

end

    activeTouch = nil
    moveDir = Vector3.new(0,0,0)
    if GrayGui and GrayGui:FindFirstChild("TouchBlocker") then
        local stick = GrayGui.TouchBlocker.MainFrame.Stick
        local half = _G_settings.JoySize / 2
        stick:TweenPosition(UDim2.new(0.5, -half/2, 0.5, -half/2), "Out", "Back", 0.1)
    end
    SetMoveVector(Vector3.new(0, 0, 0))
end

    Instance.new("UICorner", stick).CornerRadius = UDim.new(1, 0)

    GrayGui = sg
    ApplyJoystickSettings()

    main.InputBegan:Connect(function(input)
        if (input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1) then
            activeTouch = input
            local center = main.AbsolutePosition + (main.AbsoluteSize / 2)
            local diff = Vector2.new(input.Position.X - center.X, input.Position.Y - center.Y)
            local limit = main.AbsoluteSize.X / 2
            local dist = math.min(diff.Magnitude, limit)
            if dist > 0.05 then
                local dir = diff.Unit
                SetMoveVector(Vector3.new(dir.X * (dist/limit), 0, dir.Y * (dist/limit)))
            end
        end
    end)
    UserInputService.InputChanged:Connect(function(input)
        if input == activeTouch and sg.Enabled then
            local center = main.AbsolutePosition + (main.AbsoluteSize / 2)
            local diff = Vector2.new(input.Position.X - center.X, input.Position.Y - center.Y)
            local limit = main.AbsoluteSize.X / 2
            local dist = math.min(diff.Magnitude, limit)
            local dir = diff.Unit
            local half = _G_settings.JoySize / 2
            stick.Position = UDim2.new(0.5, (dir.X * dist) - half/2, 0.5, (dir.Y * dist) - half/2)
            if dist > 0.05 then SetMoveVector(Vector3.new(dir.X * (dist/limit), 0, dir.Y * (dist/limit))) else SetMoveVector(Vector3.new(0, 0, 0)) end
        end
    end)
    UserInputService.InputEnded:Connect(function(input) if input == activeTouch then stopMoving() end end)
    UserInputService.WindowFocusReleased:Connect(stopMoving)
    return sg
end

local function UpdateGrayMovementState()
    local shouldBeActive = _G_settings.CustomJoystick
    local touchGui = pgui:FindFirstChild("TouchGui")
    local function toggleDefaultThumbstick(visible)
        if touchGui then
            local frame = touchGui:FindFirstChild("TouchControlFrame")
            if frame then
                for _, v in pairs(frame:GetChildren()) do
                    if string.find(v.Name, "Thumbstick") then
                        for _, desc in pairs(v:GetDescendants()) do if desc:IsA("ImageLabel") or desc:IsA("ImageButton") then desc.ImageTransparency = visible and 0 or 1 end end
                        v.BackgroundTransparency = 1
                    end
                end
            end
        end
    end
    if shouldBeActive then
        toggleDefaultThumbstick(false)
        if not GrayGui then GrayGui = setupGrayJoy() end
        GrayGui.Enabled = true
        ApplyJoystickSettings()
    else
        toggleDefaultThumbstick(true)
        if GrayGui and GrayGui.Enabled then GrayGui.Enabled = false; stopMoving() end
    end
end

local function CreatePlayerSlider(parent, name, min, max, startVal, callback)
    local container = Instance.new("Frame", parent)
    container.BackgroundColor3 = Color3.fromRGB(20, 20, 20)
    Instance.new("UICorner", container)
    local stroke = Instance.new("UIStroke", container)
    stroke.Color = Color3.fromRGB(150, 0, 0)
    
    local label = Instance.new("TextLabel", container)
    label.Size = UDim2.new(1, -10, 0.5, 0)
    label.Position = UDim2.new(0, 5, 0, 0)
    label.BackgroundTransparency = 1
    label.Text = name .. ": " .. startVal
    label.TextColor3 = Color3.new(1, 1, 1)
    label.Font = "GothamBold"
    label.TextSize = 12
    label.TextXAlignment = Enum.TextXAlignment.Left
    
    local track = Instance.new("Frame", container)
    track.Size = UDim2.new(1, -20, 0.15, 0)
    track.Position = UDim2.new(0, 10, 0.65, 0)
    track.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
    Instance.new("UICorner", track)
    
    local fill = Instance.new("Frame", track)
    local perc = (startVal - min) / (max - min)
    fill.Size = UDim2.new(perc, 0, 1, 0)
    fill.BackgroundColor3 = Color3.fromRGB(255, 0, 0)
    Instance.new("UICorner", fill)
    
    local isDragging = false 
    
    local function updateSlider(input)
        local relativeX = input.Position.X - track.AbsolutePosition.X
        local clampedX = math.clamp(relativeX, 0, track.AbsoluteSize.X)
        local p = clampedX / track.AbsoluteSize.X
        fill.Size = UDim2.new(p, 0, 1, 0)
        local val = math.floor(min + (max - min) * p)
        label.Text = name .. ": " .. val
        callback(val)
    end
    
    track.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            isDragging = true
            updateSlider(input)
        end
    end)
    
    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            isDragging = false
        end
    end)
    
    UserInputService.InputChanged:Connect(function(input)
        if isDragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
            updateSlider(input)
        end
    end)
end

local ActiveEditorSlider = false 
local function CreateEditorSlider(parentUI, name, yPos, min, max, startVal, callback)
    local label = Instance.new("TextLabel", parentUI)
    label.Size = UDim2.new(1, -20, 0, 20)
    label.Position = UDim2.new(0, 10, 0, yPos)
    label.BackgroundTransparency = 1
    label.Text = name .. ": " .. math.floor(startVal)
    label.TextColor3 = Color3.new(1, 1, 1)
    label.Font = "GothamBold"
    label.TextSize = 12
    label.TextXAlignment = Enum.TextXAlignment.Left

    local container = Instance.new("Frame", parentUI)
    container.Size = UDim2.new(1, -20, 0, 20)
    container.Position = UDim2.new(0, 10, 0, yPos + 20)
    container.BackgroundColor3 = Color3.fromRGB(25, 25, 25)
    Instance.new("UICorner", container)
   
    local stroke = Instance.new("UIStroke", container)
    stroke.Color = Color3.fromRGB(100, 100, 100)

    local fill = Instance.new("Frame", container)
    local perc = (startVal - min) / (max - min)
    fill.Size = UDim2.new(perc, 0, 1, 0)
    fill.BackgroundColor3 = ScaryEdgeColor
    Instance.new("UICorner", fill)

    local isDragging = false
    container.InputBegan:Connect(function(input)
        if not ActiveEditorSlider and (input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch) then 
            isDragging = true 
            ActiveEditorSlider = true
        end
    end)
    fill.InputBegan:Connect(function(input)
        if not ActiveEditorSlider and (input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch) then 
            isDragging = true 
            ActiveEditorSlider = true
        end
    end)
 
    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then 
            if isDragging then
                isDragging = false 
                ActiveEditorSlider = false
            end
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if isDragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) and parentUI.Visible then
            local relativeX = input.Position.X - container.AbsolutePosition.X
            local clampedX = math.clamp(relativeX, 0, container.AbsoluteSize.X)
            local p = clampedX / container.AbsoluteSize.X
            fill.Size = UDim2.new(p, 0, 1, 0)
            local val = min + (max - min) * p
            label.Text = name .. ": " .. math.floor(val)
            callback(val)
        end
    end)

    return function(newVal)
        local p = (newVal - min) / (max - min)
        fill.Size = UDim2.new(p, 0, 1, 0)
        label.Text = name .. ": " .. math.floor(newVal)
    end
end

local function HideAllPages()
    SysP.Visible = false
    BotP.Visible = false
    PlayerP.Visible = false
    CredP.Visible = false
end

UI_Elements.Sys.MouseButton1Click:Connect(function() HideAllPages(); SysP.Visible=true end)
UI_Elements.Bot.MouseButton1Click:Connect(function() HideAllPages(); BotP.Visible=true end)
UI_Elements.Player.MouseButton1Click:Connect(function() HideAllPages(); PlayerP.Visible=true end)
UI_Elements.Cred.MouseButton1Click:Connect(function() HideAllPages(); CredP.Visible=true end)

local function StartSequence()
    music:Play()
    TweenService:Create(music, TweenInfo.new(1.5), {Volume = 1}):Play()
    MusicBtn.Visible = true 
    LoadingFrame.Visible = true
    IntroFrame.Visible = true;
    IntroFrame.BackgroundTransparency = 1; IntroLogo.ImageTransparency = 1; WelcomeLabel.TextTransparency = 1; WelcomeShadow.TextTransparency = 1
    LoadLabel.TextTransparency = 1
    local fadeTime = TweenInfo.new(1, Enum.EasingStyle.Quart)
    TweenService:Create(IntroFrame, fadeTime, {BackgroundTransparency = 0}):Play()
    TweenService:Create(BlurEffect, fadeTime, {Size = 24}):Play();
    task.wait(1)
    TweenService:Create(WelcomeLabel, fadeTime, {TextTransparency = 0}):Play(); TweenService:Create(WelcomeShadow, fadeTime, {TextTransparency = 0}):Play()
    TweenService:Create(LoadLabel, fadeTime, {TextTransparency = 0}):Play()
    
    task.spawn(function()
        local s = tick()
        while tick() - s < 4 do
            local pulse = (math.sin(tick() * 4) + 1) / 2
            local c = Color3.fromRGB(100, 0, 0):Lerp(Color3.fromRGB(255, 50, 50), pulse)
            WelcomeLabel.TextColor3 = c; WelcomeShadow.TextColor3 = c; LoadLabel.TextColor3 = c
            local dots = string.rep(".", math.floor(tick() * 3.5) % 4)
            LoadLabel.Text = "Loading" .. dots; task.wait()
        end
    end)
    
    task.wait(4)
    TweenService:Create(WelcomeLabel, fadeTime, {TextTransparency = 1}):Play()
    TweenService:Create(WelcomeShadow, fadeTime, {TextTransparency = 1}):Play();
    TweenService:Create(LoadLabel, fadeTime, {TextTransparency = 1}):Play()
    task.wait(1) 
    TweenService:Create(IntroLogo, TweenInfo.new(1.2), {ImageTransparency = 0}):Play()
    task.wait(3) 
    TweenService:Create(IntroLogo, TweenInfo.new(1.5), {ImageTransparency = 1}):Play();
    TweenService:Create(IntroFrame, TweenInfo.new(1.5), {BackgroundTransparency = 1}):Play(); TweenService:Create(BlurEffect, TweenInfo.new(1.5), {Size = 0}):Play()
    task.wait(1.5)
    LoadingFrame:Destroy();
    IntroFrame:Destroy()
    CustomBtn.Visible = true
    Refresh()
end

local LastFPSUpdate = tick()
local FrameCount = 0

RunService.RenderStepped:Connect(function(deltaTime)
    FrameCount = FrameCount + 1
    if tick() - LastFPSUpdate >= 0.5 then
        if _G_settings.ShowFPSCounter then FPSDisplay.Text = "FPS: " .. math.floor(FrameCount / (tick() - LastFPSUpdate)) end
        FrameCount = 0; LastFPSUpdate = tick()
    end
end)

task.spawn(StartSequence)
