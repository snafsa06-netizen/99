RunService.Heartbeat:Connect(function(deltaTime)
    local char = lp.Character
    if not char then return end
    local hum = char:FindFirstChildOfClass("Humanoid")
    local root = char:FindFirstChild("HumanoidRootPart")
    
    if hum and root and hum.Health > 0 then
        if NoStopEnabled then
            if hum.PlatformStand then hum.PlatformStand = false end
            if hum.AutoRotate == false then hum.AutoRotate = true end
            if root.Anchored and not (Settings.MeteorFarmBiggest or Settings.MeteorFarmClosest or Settings.DiamondFarm or isEatingOrDrinking) then root.Anchored = false end
        end
        if isSpeedForced and forceSpeed > 16 and not (Settings.MeteorFarmBiggest or Settings.MeteorFarmClosest or Settings.DiamondFarm or isEatingOrDrinking) then
            hum.WalkSpeed = forceSpeed
        end
        if isJumpForced then
            hum.UseJumpPower = true
            hum.JumpPower = forceJump
        end
    end
end)

local isAutoPlayRunning = false
local function runAutoPlay()
    if isAutoPlayRunning then return end
    isAutoPlayRunning = true
    task.spawn(function()
        while Settings.AutoPlay do
            task.wait(5)

local function pressButton(btn, extraXOffset)
    extraXOffset = extraXOffset or 0
    if btn and btn.AbsoluteSize.X > 0 and btn.Visible then
        local visualX = btn.AbsolutePosition.X + (btn.AbsoluteSize.X / 2)
        local visualY = btn.AbsolutePosition.Y + (btn.AbsoluteSize.Y / 2)
        local parentGui = btn:FindFirstAncestorOfClass("ScreenGui")
        if parentGui and not parentGui.IgnoreGuiInset then
            local inset = GuiService:GetGuiInset()
            visualX = visualX + inset.X
            visualY = visualY + inset.Y
        end
        if visualX <= 10 or visualY <= 10 then return end
        pcall(function()
            local hardwareClickX = visualX + VIM_X_OFFSET + extraXOffset
            local hardwareClickY = visualY + VIM_Y_OFFSET
            vim:SendMouseButtonEvent(hardwareClickX, hardwareClickY, 0, true, game, 1)
            task.wait(0.03)
            vim:SendMouseButtonEvent(hardwareClickX, hardwareClickY, 0, false, game, 1)
        end)
    end
end

local function monitorHealth(character)
    if not character then return end
    local humanoid = character:WaitForChild("Humanoid", 10)
    if humanoid then
        if humanoid.Health <= 0 then
            resetSkippedCache()
            if Settings.AutoRespawn then runRespawnSequence() end
        end
        humanoid.HealthChanged:Connect(function(currentHealth)
            if currentHealth <= 0 then
                resetSkippedCache()
                if Settings.AutoRespawn then runRespawnSequence() end
            end
        end)
        humanoid.Died:Connect(function()
            resetSkippedCache()
            if Settings.AutoRespawn then runRespawnSequence() end
        end)
    end
end

local FLY_SPEED = 60           
local NEXT_DELAY = 0.05         
local EXPLOSION_TRAVEL_TIMEOUT = 40 

local function absoluteFlyTo(hrp, humanoid, targetPosition, maxTimeLeft, stopDistance, isFoodFlight)
    stopDistance = stopDistance or 6
    if not hrp or not hrp.Parent or not humanoid then return false end
    local startPos = hrp.Position
    local totalDistance = (targetPosition - startPos).Magnitude
    local flyTime = totalDistance / FLY_SPEED
    local startTime = os.clock()
    local reached = false

    if humanoid then humanoid.PlatformStand = true end

    while true do
        if not hrp or not hrp.Parent or humanoid.Health <= 0 then break end
        if not (Settings.MeteorFarmBiggest or Settings.MeteorFarmClosest or Settings.DiamondFarm or isEatingOrDrinking) then break end
        if not isFoodFlight and isEatingOrDrinking then break end

        local elapsed = os.clock() - startTime
        if elapsed > maxTimeLeft then break end

        local currentDist = (targetPosition - hrp.Position).Magnitude
        if currentDist <= stopDistance then reached = true; break end

        local alpha = math.clamp(elapsed / flyTime, 0, 1)
        local nextPos = startPos:Lerp(targetPosition, alpha)
        hrp.CFrame = CFrame.new(nextPos, targetPosition)

        if alpha >= 1 then reached = true; break end
        RunService.Heartbeat:Wait()
    end

    if humanoid then humanoid.PlatformStand = false end
    if hrp and hrp.Parent then
        hrp.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
        hrp.AssemblyAngularVelocity = Vector3.new(0, 0, 0)
    end
    return reached
end

local function getClosestLeaf(hrp)
    local closest = nil
    local minDist = math.huge
    for _, v in pairs(workspace:GetDescendants()) do
        if v:IsA("BasePart") and v.Name == "Leaf" then
            local dist = (v.Position - hrp.Position).Magnitude
            if dist < minDist then minDist = dist; closest = v end
        end
    end
    return closest
end

local function getClosestWater(hrp)
    local closest = nil
    local minDist = math.huge
    for _, v in pairs(workspace:GetDescendants()) do
        if v:IsA("BasePart") and v.Name == "Water" then
            local dist = (v.Position - hrp.Position).Magnitude
            if dist < minDist then minDist = dist; closest = v end
        end
    end
    return closest
end

local function processConsumable(hrp, hum, typeStr, duration)
    local elapsed = 0
    local reachedDest = false
    isEatingOrDrinking = true
    
    while elapsed < duration and (Settings.FoodFarm or Settings.WaterFarm) do
        if hum.Health <= 0 then break end
        local target = nil
        local stopDist = 5
        if typeStr == "Food" and Settings.FoodFarm then
            target = getClosestLeaf(hrp); stopDist = 13
        elseif typeStr == "Water" and Settings.WaterFarm then
            target = getClosestWater(hrp); stopDist = 8
        else break end
        
        if not target or not target.Parent then task.wait(0.5); continue end
        
        absoluteFlyTo(hrp, hum, target.Position, 10, stopDist, true)
        reachedDest = true
        hrp.Anchored = true
        
        local oldMin = lp.CameraMinZoomDistance
        local oldMax = lp.CameraMaxZoomDistance
        lp.CameraMinZoomDistance = 0.5
        lp.CameraMaxZoomDistance = 0.5
        
        local cam = workspace.CurrentCamera
        local cx, cy = cam.ViewportSize.X / 2, cam.ViewportSize.Y / 2
        
        while elapsed < duration and target and target.Parent and hum.Health > 0 do
            cam.CFrame = CFrame.new(cam.CFrame.Position, target.Position)
            pcall(function()
                vim:SendMouseButtonEvent(cx, cy, 0, true, game, 1)
                task.wait(0.01)
                vim:SendMouseButtonEvent(cx, cy, 0, false, game, 1)
            end)
            task.wait(0.05)
            elapsed = elapsed + 0.06
        end
        
        lp.CameraMinZoomDistance = oldMin
        lp.CameraMaxZoomDistance = oldMax
        hrp.Anchored = false
    end
    
    return reachedDest
end

task.spawn(function()
    while task.wait(1) do
        local char = lp.Character
        local hrp = char and char:FindFirstChild("HumanoidRootPart")
        local hum = char and char:FindFirstChild("Humanoid")
        
        if hrp and hum and hum.Health > 0 then
            if Settings.FoodFarm or Settings.WaterFarm then
                local cycleDone = false
                local savedPos = hrp.Position
                
                if Settings.FoodFarm then
                    local didReach = processConsumable(hrp, hum, "Food", 5)
                    if didReach then cycleDone = true end
                end
                
                if Settings.WaterFarm then
                    local didReach = processConsumable(hrp, hum, "Water", 10)
                    if didReach then cycleDone = true end
                end
                
                if cycleDone then
                    isEatingOrDrinking = true
                    absoluteFlyTo(hrp, hum, savedPos, 30, 5, true)
                    hrp.CFrame = CFrame.new(savedPos)
                    isEatingOrDrinking = false
                    
                    local waitTime = 60
                    while waitTime > 0 and (Settings.FoodFarm or Settings.WaterFarm) do
                        task.wait(1); waitTime = waitTime - 1
                    end
                end
            end
        end
    end
end)

local function cacheObject(obj)
    if obj:IsA("Part") and obj.Name == "Explosion" then
        local isOfficialExplosion = false
        pcall(function()
            if obj.Material == Enum.Material.Glass and obj.BrickColor.Name == "Steel blue" and obj.Size.X >= 90 then
                isOfficialExplosion = true
            end
        end)
        if isOfficialExplosion then activeExplosions[obj] = true end
    elseif obj:IsA("Model") and obj.Name == "Meteor" then
        activeMeteors[obj] = true
    elseif obj:IsA("Model") and (string.find(obj.Name, "Diamond Rock") or obj:GetAttribute("DiamondRock")) then
        activeDiamonds[obj] = true
    end
end

for _, obj in pairs(workspace:GetDescendants()) do cacheObject(obj) end
workspace.DescendantAdded:Connect(cacheObject)

workspace.DescendantRemoving:Connect(function(obj)
    if activeExplosions[obj] then activeExplosions[obj] = nil; skippedExplosions[obj] = nil
    elseif activeMeteors[obj] then activeMeteors[obj] = nil; skippedMeteors[obj] = nil
    elseif activeDiamonds[obj] then activeDiamonds[obj] = nil; skippedDiamonds[obj] = nil end
end)

local function getValidMeteorCount()
    local count = 0
    for meteor in pairs(activeMeteors) do
        if meteor and meteor.Parent then count = count + 1 else activeMeteors[meteor] = nil end
    end
    return count
end

local function getValidDiamondCount()
    local count = 0
    for diamond in pairs(activeDiamonds) do
        if diamond and diamond.Parent then count = count + 1 else activeDiamonds[diamond] = nil end
    end
    return count
end

local function skipExplosionZone(centerPosition)
    for p in pairs(activeExplosions) do
        pcall(function() 
            if p and p.Parent and (p.Position - centerPosition).Magnitude < 100 then skippedExplosions[p] = true end
        end)
    end
end

local function getClosestExplosion(hrp)
    local closest = nil; local minDistance = math.huge; local hrpPos = hrp.Position 
    for obj in pairs(activeExplosions) do
        local success = pcall(function()
            if obj and obj.Parent and not skippedExplosions[obj] then
                local dist = (hrpPos - obj.Position).Magnitude
                if dist < minDistance then minDistance = dist; closest = obj end
            elseif not obj or not obj.Parent then activeExplosions[obj] = nil; skippedExplosions[obj] = nil end
        end)
        if not success then activeExplosions[obj] = nil end
    end
    return closest
end

local function getBiggestMeteor()
    local biggestMeteor = nil; local maxSize = -1
    for obj in pairs(activeMeteors) do
        pcall(function()
            if obj and obj.Parent and not skippedMeteors[obj] then
                local extentsSize = obj:GetExtentsSize()
                local sizeVolume = extentsSize.X * extentsSize.Y * extentsSize.Z
                if sizeVolume > maxSize then maxSize = sizeVolume; biggestMeteor = obj end
            elseif not obj or not obj.Parent then activeMeteors[obj] = nil; skippedMeteors[obj] = nil end
        end)
    end
    return biggestMeteor
end

local function startContinuousPlatform(hrp)

        local platform = Instance.new("Part")

        platform.Anchored = true
        platform.Transparency = 1 
        platform.Color = Color3.fromRGB(0, 255, 128)
        platform.Material = Enum.Material.Neon
        platform.Parent = workspace
        activeFakeFloor = platform
    end
    if currentPlatformHRP ~= hrp then
        currentPlatformHRP = hrp
        if platformConnection then platformConnection:Disconnect() end
        platformConnection = RunService.Heartbeat:Connect(function()
            if activeFakeFloor and activeFakeFloor.Parent and hrp and hrp.Parent then
                activeFakeFloor.CFrame = CFrame.new(hrp.Position.X, hrp.Position.Y - 3.6, hrp.Position.Z)
            end
        end)
    end
end

local function SafeTextScale(obj, optimalSize)
    if obj:IsA("TextLabel") or obj:IsA("TextButton") or obj:IsA("TextBox") then
        local maxScaleFactor = 2.0
        local targetSize = math.floor(optimalSize * maxScaleFactor)
        obj.TextScaled = true

        local constraint = obj:FindFirstChildOfClass("UITextSizeConstraint")
        if not constraint then
            constraint = Instance.new("UITextSizeConstraint")
            constraint.Parent = obj
        end
        constraint.MaxTextSize = targetSize
        constraint.MinTextSize = 6
    end
end

                if not targetExplosion and Settings.DiamondFarm then targetDiamond = getClosestDiamond(hrp) end
                
                local isMeteorBanned = false
                if not targetExplosion and not targetDiamond and (Settings.MeteorFarmBiggest or Settings.MeteorFarmClosest) then
                    local mCount = getValidMeteorCount()
                    if mCount > 10 then 
                        if Settings.MeteorFarmClosest then targetMeteor = getClosestMeteor(hrp)
                        elseif Settings.MeteorFarmBiggest then targetMeteor = getBiggestMeteor() end
                    else isMeteorBanned = true end
                end

                if isMeteorBanned and not targetDiamond and not targetExplosion then
                    if not isWarningVisible then
                        isWarningVisible = true
                        TweenService:Create(AntiBanWarning, TweenInfo.new(0.5), {BackgroundTransparency = 0.90, TextTransparency = 0}):Play()
                    end
                    if humanoid then humanoid.PlatformStand = true end 
                    if hrp then 
                        hrp.AssemblyLinearVelocity = Vector3.new(0,0,0)
                        hrp.AssemblyAngularVelocity = Vector3.new(0,0,0) 
                        hrp.Anchored = true 
                    end
                    return 
                else
                    if isWarningVisible then
                        isWarningVisible = false
                        TweenService:Create(AntiBanWarning, TweenInfo.new(0.5), {BackgroundTransparency = 1, TextTransparency = 1}):Play()
                    end
                end

                if not targetExplosion and not targetDiamond and not targetMeteor then 
                    resetSkippedCache()
                    if humanoid then humanoid.PlatformStand = true end 
                    if hrp then 
                        hrp.AssemblyLinearVelocity = Vector3.new(0,0,0)
                        hrp.AssemblyAngularVelocity = Vector3.new(0,0,0) 
                        hrp.Anchored = true 
                    end
                    return 
                else
                    if hrp then hrp.Anchored = false end 
                end
                 
                if targetExplosion then
                    local savedExplosionPos = targetExplosion.Position
                    local landPos = Vector3.new(targetExplosion.Position.X, targetExplosion.Position.Y, targetExplosion.Position.Z)
                    local reached = absoluteFlyTo(hrp, humanoid, landPos, EXPLOSION_TRAVEL_TIMEOUT, 6, false)
                    if not reached then
                        if not isEatingOrDrinking then skipExplosionZone(savedExplosionPos) end
                        return
                    end
                    if targetExplosion and targetExplosion.Parent and hrp and hrp.Parent then
                        hrp.CFrame = targetExplosion.CFrame
                        hrp.Anchored = true 
                        local collectStart = os.clock()
                        while os.clock() - collectStart < 7.0 and humanoid.Health > 0 and isFarmingAnything do
                            if isEatingOrDrinking then break end 
                            if not targetExplosion or not targetExplosion.Parent then break end
                            if firetouchinterest then
                                pcall(function()
                                    firetouchinterest(hrp, targetExplosion, 0)
                                    task.wait(0.01) 
                                    if targetExplosion and targetExplosion.Parent then firetouchinterest(hrp, targetExplosion, 1) end
                                end)
                            end
                            task.wait(0.05)
                        end
                        if hrp and hrp.Parent then hrp.Anchored = false end
                        skipExplosionZone(savedExplosionPos)
                        task.wait(NEXT_DELAY)
                    end
                elseif targetDiamond then
                    local landPos = targetDiamond:GetPivot().Position
                    local reached = absoluteFlyTo(hrp, humanoid, landPos, 30, 5, false)
                    if not reached then return end
                    local collectStart = os.clock()
                    while os.clock() - collectStart < 4.0 and humanoid.Health > 0 and Settings.DiamondFarm do
                        if isEatingOrDrinking then break end
                        if not targetDiamond or not targetDiamond.Parent then break end
                        task.wait(0.1)
                    end
                    skippedDiamonds[targetDiamond] = true
                    task.wait(NEXT_DELAY)
                elseif targetMeteor then
                    local landPos = targetMeteor:GetPivot().Position
                    local reached = absoluteFlyTo(hrp, humanoid, landPos, 60, 60, false)
                    if not reached then return end
                    task.wait(0.5) 
                    skippedMeteors[targetMeteor] = true
                    task.wait(NEXT_DELAY)
                end
            end)
        else
            if isWarningVisible then
                isWarningVisible = false
                TweenService:Create(AntiBanWarning, TweenInfo.new(0.5), {BackgroundTransparency = 1, TextTransparency = 1}):Play()
            end
            if not isEatingOrDrinking then
                if activeFakeFloor then activeFakeFloor:Destroy(); activeFakeFloor = nil; currentPlatformHRP = nil end
                if platformConnection then platformConnection:Disconnect(); platformConnection = nil end
                
                local character = lp.Character
                if character then
                    local hrp = character:FindFirstChild("HumanoidRootPart")
                    local humanoid = character:FindFirstChild("Humanoid")
                    if hrp then hrp.Anchored = false end
                    if humanoid then humanoid.PlatformStand = false end
                end
            end
        end
    end
end)

local function GetDynamicScale()
    local viewport = Camera.ViewportSize
    local ratio = 1
    if viewport.X > 0 and viewport.Y > 0 then
        ratio = math.min(viewport.X / 1280, viewport.Y / 720)
    end
    local scaleVal = 0.5 + (SizeIndex * 0.1)
    return ratio * scaleVal
end

local function ApplyResponsiveScale(guiElement)
    local uiScale = Instance.new("UIScale", guiElement)
    local function UpdateScale()
        if Camera then
            local viewport = Camera.ViewportSize
            local baseRes = Vector2.new(1280, 720)
            local scale = math.min(viewport.X / baseRes.X, viewport.Y / baseRes.Y)
            uiScale.Scale = math.clamp(scale * 1.3, 1.0, 6.25)
        end
    end
    Camera:GetPropertyChangedSignal("ViewportSize"):Connect(UpdateScale)
    UpdateScale()
end

local function AnimateButton(btn, clickCallback, isColorSquare)
    local uiScale = btn:FindFirstChildOfClass("UIScale") or Instance.new("UIScale", btn)
    local tweenInfoSlow = TweenInfo.new(0.22, Enum.EasingStyle.Sine, Enum.EasingDirection.Out)
    local tweenInfoPress = TweenInfo.new(0.1, Enum.EasingStyle.Sine, Enum.EasingDirection.Out)
    local activeScaleTween = nil
    local activeBtnTween = nil
    local function playTween(targetScale, targetBgTrans, targetTextTrans, speedInfo)
        if activeScaleTween then activeScaleTween:Cancel() end
        if activeBtnTween then activeBtnTween:Cancel() end
        activeScaleTween = TweenService:Create(uiScale, speedInfo, {Scale = targetScale})
        activeBtnTween = TweenService:Create(btn, speedInfo, {BackgroundTransparency = targetBgTrans, TextTransparency = targetTextTrans})
        activeScaleTween:Play()
        activeBtnTween:Play()
    end
    local isImage = btn:IsA("ImageButton")
    local baseBg = isImage and 1 or 0.90
    local hoverBg = isImage and 1 or 0.96
    local pressBg = isImage and 1 or 1.0
    btn.AutoButtonColor = false
    if not isImage then
        btn.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
        btn.BackgroundTransparency = baseBg
    end
    btn.MouseEnter:Connect(function()
        local isActive = btn:GetAttribute("IsActive")
        playTween(isActive and 1.10 or 1.04, isActive and 1 or hoverBg, 0, tweenInfoSlow)
    end)
    btn.MouseLeave:Connect(function()
        local isActive = btn:GetAttribute("IsActive")
        playTween(isActive and 1.10 or 1.0, isActive and 1 or baseBg, 0, tweenInfoSlow)
    end)
    btn.MouseButton1Down:Connect(function()
        local isActive = btn:GetAttribute("IsActive")
        playTween(isActive and 1.05 or 0.94, pressBg, 0.1, tweenInfoPress)
    end)
    btn.MouseButton1Up:Connect(function()
        local isActive = btn:GetAttribute("IsActive")
        playTween(isActive and 1.10 or 1.04, hoverBg, 0, tweenInfoSlow)
    end)
    btn.MouseButton1Click:Connect(function()
        ClickSound:Play()
        if clickCallback then clickCallback() end
    end)
end

local GlowContainer = Instance.new("Frame")
GlowContainer.Name = "GlowContainer"
GlowContainer.Parent = MainFrame
GlowContainer.BackgroundTransparency = 1
GlowContainer.Size = UDim2.new(1, 0, 1, 0)
GlowContainer.AnchorPoint = Vector2.new(0.5, 0.5)
GlowContainer.Position = UDim2.new(0.5, 0, 0.5, 0)
GlowContainer.ZIndex = 3

local TotalLayers = 150
local LayersData = {}
local ZoomRatio = 2

for i = 1, TotalLayers do
    local Layer = Instance.new("Frame")
    Layer.Parent = GlowContainer
    Layer.AnchorPoint = Vector2.new(0.5, 0.5)
    Layer.BorderSizePixel = 0
    Layer.ZIndex = 4
    local baseSize = (40 + (i * 2.5)) * ZoomRatio
    Layer.Size = UDim2.new(0, baseSize, 0, baseSize)
    local progress = (TotalLayers - i) / TotalLayers
    Layer.BackgroundTransparency = 1
    Instance.new("UICorner", Layer).CornerRadius = UDim.new(1, 0)
    table.insert(LayersData, { Instance = Layer, BaseSize = baseSize, OffsetFactor = (i / TotalLayers), Progress = progress })
end

task.spawn(function()
    local timeCounter = 0
    while true do
        timeCounter = timeCounter + 0.007
        CurrentGlowColor = CurrentGlowColor:Lerp(TargetGlowColor, 0.03)
        for _, data in pairs(LayersData) do
            if data.Instance and data.Instance.Parent then
                local offsetX, offsetY, sizePulse
                if CurrentAnimStyle == "1v" then
                    local waveAngle = timeCounter * 0.6 + (data.OffsetFactor * math.pi * 1.5)
                    offsetX = math.sin(waveAngle) * 120
                    offsetY = math.cos(waveAngle * 0.5) * 40
                    sizePulse = 1 + (math.sin(waveAngle * 0.8) * 0.12)
                elseif CurrentAnimStyle == "2v" then
                    local waveAngle = timeCounter * 1.2 + (data.OffsetFactor * math.pi * 2.5)
                    offsetX = math.sin(waveAngle) * 80
                    offsetY = math.cos(waveAngle * 0.8) * 80
                    sizePulse = 1 + (math.sin(timeCounter * 1.5 + data.OffsetFactor) * 0.25)
                elseif CurrentAnimStyle == "3v" then
                    local speed = 0.15 + (data.OffsetFactor * 0.35)
                    local seed = data.OffsetFactor * 100
                    local layerAngle = timeCounter * (speed * 1) + seed
                    offsetX = (math.sin(layerAngle) * 40) - 50
                    offsetY = math.cos(layerAngle * 0.8) * 20
                    sizePulse = 1 + (math.sin(layerAngle * 1.2) * 0.25)
                end
                local distanceFactor = 0.8 * ZoomRatio * 1.5
                data.Instance.Position = UDim2.new(0.5, offsetX * distanceFactor, 0.5, offsetY * distanceFactor)
                local currentSize = data.BaseSize * sizePulse
                data.Instance.Size = UDim2.new(0, currentSize, 0, currentSize)
                local baseTrans = 0.983 + (data.Progress * 0.008)
                local finalTrans = 1 - ((1 - baseTrans) * GlowAlpha)
                data.Instance.BackgroundTransparency = finalTrans
                data.Instance.BackgroundColor3 = CurrentGlowColor
            end
        end
        task.wait()
    end
end)

local function ApplyColorTheme(glowColor, activeBtn)
    TargetGlowColor = glowColor
    local r, g, b = glowColor.R, glowColor.G, glowColor.B
    local mainColor = Color3.new(r * 0.14, g * 0.14, b * 0.14)
    local boxColor  = Color3.new(r * 0.08, g * 0.08, b * 0.08)
    CurrentMainBgColor = mainColor
    CurrentBoxBgColor  = boxColor
    local ti = TweenInfo.new(0.8, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
    for _, box in pairs(AllUIBoxes) do
        if box and box:IsA("Frame") and box.Name == "MainFrame" then
            TweenService:Create(box, ti, {BackgroundColor3 = mainColor}):Play()
        end
    end
    -- تحديث الزر النشط
    if activeBtn then
        if ActiveColorBtn then
            ActiveColorBtn.Name = "ColorBtn"
            local oldStroke = ActiveColorBtn:FindFirstChildOfClass("UIStroke")
            if oldStroke then TweenService:Create(oldStroke, ti, {Transparency = 1}):Play() end
        end
        ActiveColorBtn = activeBtn
        ActiveColorBtn.Name = "ActiveColorBtn"
        local newStroke = ActiveColorBtn:FindFirstChildOfClass("UIStroke")
        if newStroke then TweenService:Create(newStroke, ti, {Transparency = 0}):Play() end
    end
    -- تحديث visual لكل أزرار الألوان
    if _G.ColorBtnsList then
        for _, btn in ipairs(_G.ColorBtnsList) do
            local scaleObj = btn:FindFirstChildOfClass("UIScale")
            if btn == ActiveColorBtn then
                if scaleObj then TweenService:Create(scaleObj, TweenInfo.new(0.3, Enum.EasingStyle.Sine, Enum.EasingDirection.Out), {Scale = 1.15}):Play() end
                TweenService:Create(btn, TweenInfo.new(0.3), {BackgroundTransparency = 0.0}):Play()
                btn:SetAttribute("TargetScale", 1.15)
                btn:SetAttribute("TargetTrans", 0.0)
            else
                if scaleObj then TweenService:Create(scaleObj, TweenInfo.new(0.3, Enum.EasingStyle.Sine, Enum.EasingDirection.Out), {Scale = 0.85}):Play() end
                TweenService:Create(btn, TweenInfo.new(0.3), {BackgroundTransparency = 0.4}):Play()
                btn:SetAttribute("TargetScale", 0.85)
                btn:SetAttribute("TargetTrans", 0.4)
            end
        end
    end
end

local function CreatePlayerSlider(parent, name, min, max, startVal, callback)
    local container = Instance.new("Frame", parent)
    container.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
    container.BackgroundTransparency = 0.90
    Instance.new("UICorner", container).CornerRadius = UDim.new(0, 4)

    local label = Instance.new("TextLabel", container)
    label.Size = UDim2.new(1, -10, 0, 25)
    label.Position = UDim2.new(0, 5, 0, 5)
    label.BackgroundTransparency = 1
    label.Text = name .. ": " .. startVal
    label.TextColor3 = Color3.new(1, 1, 1)
    label.Font = Enum.Font.Gotham
    SafeTextScale(label, 8)
    label.TextXAlignment = Enum.TextXAlignment.Left

    local sliderBg = Instance.new("Frame", container)
    sliderBg.Size = UDim2.new(1, -20, 0, 8)
    sliderBg.Position = UDim2.new(0, 10, 0, 32)
    sliderBg.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
    sliderBg.BackgroundTransparency = 0.80
    sliderBg.BorderSizePixel = 0
    Instance.new("UICorner", sliderBg).CornerRadius = UDim.new(1, 0)

    local fill = Instance.new("Frame", sliderBg)
    local perc = (startVal - min) / (max - min)
    fill.Size = UDim2.new(perc, 0, 1, 0)
    fill.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
    fill.BackgroundTransparency = 0.80
    Instance.new("UICorner", fill).CornerRadius = UDim.new(1, 0)

    local isDragging = false
    local function updateSlider(input)
        local relativeX = input.Position.X - sliderBg.AbsolutePosition.X
        local clampedX = math.clamp(relativeX, 0, sliderBg.AbsoluteSize.X)
        local p = clampedX / sliderBg.AbsoluteSize.X
        fill.Size = UDim2.new(p, 0, 1, 0)
        local val = math.floor(min + (max - min) * p)
        label.Text = name .. ": " .. val
        callback(val)
    end

    sliderBg.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            isDragging = true; updateSlider(input)
        end
    end)
    fill.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            isDragging = true; updateSlider(input)
        end
    end)
    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            isDragging = false
        end
    end)
    UserInputService.InputChanged:Connect(function(input)
        if isDragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
            updateSlider(input)
        end
    end)
end

local function _saveHopIndex(idx)
    pcall(function()
        writefile("smarthop_state.json", HttpService:JSONEncode({index = idx, time = os.time()}))
    end)
end

local function _loadHopIndex()
    local ok, data = pcall(function()
        return HttpService:JSONDecode(readfile("smarthop_state.json"))
    end)
    if ok and data and data.index and data.time then
        if (os.time() - data.time) <= 1800 then -- 30 دقيقة
            return data.index
        end
    end
    return 1
end

local _smartHopServerIndex = _loadHopIndex()

local function _fetchServers(callback)
    task.spawn(function()
        local ok, result = pcall(function()
            return HttpService:JSONDecode(game:HttpGet("https://games.roblox.com/v1/games/"..game.PlaceId.."/servers/Public?sortOrder=Desc&excludeFullGames=true&limit=100"))
        end)
        if ok and result and result.data then
            local list = {}
            for _, srv in ipairs(result.data) do
                if srv.playing < srv.maxPlayers and srv.id ~= game.JobId then
                    table.insert(list, srv.id)
                end
            end
            callback(list)
        else
            callback({})
        end
    end)
end

local function _doHop(serverId)
    pcall(function()
        local queue_on_tp = queue_on_teleport or queueonteleport or (syn and syn.queue_on_teleport) or (fluxus and fluxus.queue_on_teleport)
        if queue_on_tp then
            queue_on_tp('task.wait(2); loadstring(game:HttpGet("https://raw.githubusercontent.com/snafsa06-netizen/99/refs/heads/main/acrocan_v16_mod.txt)()')
        end
    end)
    task.wait(0.5)
    TeleportService:TeleportToPlaceInstance(game.PlaceId, serverId, lp)
end

local function _startSmartHop(hopType)
    if _smartHopThread then task.cancel(_smartHopThread) end
    _smartHopActive = true
    _smartHopType = hopType
    _smartHopServers = {}
    _smartHopServerIndex = 1

    ShowModeMsg("Smart Hop: Fetching servers...")
    _fetchServers(function(list)
        _smartHopServers = list
        if #list == 0 then ShowModeMsg("Smart Hop: No servers found!"); return end
        ShowModeMsg("Smart Hop: Monitoring Server 1...")

        _smartHopThread = task.spawn(function()
            while _smartHopActive do
                task.wait(1)
                if not _smartHopActive then break end

                if hopType == "meteor" then
                    local count = getValidMeteorCount()
                    if count >= 10 then
                        -- نيازك كافية، ابقَ هنا
                    else
                        -- نيازك خلصت، روح للسيرفر التالي
                        _smartHopServerIndex = _smartHopServerIndex + 1
                        if _smartHopServerIndex > #_smartHopServers then
                            -- أعد جلب السيرفرات
                            ShowModeMsg("Smart Hop: Refreshing server list...")
                            _fetchServers(function(newList)
                                _smartHopServers = newList
                                _smartHopServerIndex = 1
                            end)
                            task.wait(3)
                        end
                        if _smartHopServers[_smartHopServerIndex] then
                            ShowModeMsg("Smart Hop: Hopping to server ".._smartHopServerIndex.."...")
                            _saveHopIndex(_smartHopServerIndex)
                            task.wait(1)
                            _doHop(_smartHopServers[_smartHopServerIndex])
                        end
                    end

                elseif hopType == "diamond" then
                    local count = getValidDiamondCount()
                    if count == 0 then
                        -- دايموند خلص، روح للسيرفر التالي
                        _smartHopServerIndex = _smartHopServerIndex + 1
                        if _smartHopServerIndex > #_smartHopServers then
                            ShowModeMsg("Smart Hop: Refreshing server list...")
                            _fetchServers(function(newList)
                                _smartHopServers = newList
                                _smartHopServerIndex = 1
                            end)
                            task.wait(3)
                        end
                        if _smartHopServers[_smartHopServerIndex] then
                            ShowModeMsg("Smart Hop: Hopping to server ".._smartHopServerIndex.."...")
                            _saveHopIndex(_smartHopServerIndex)
                            task.wait(1)
                            _doHop(_smartHopServers[_smartHopServerIndex])
                        end
                    end
                end
            end
        end)
    end)
end

    local btn = Instance.new("TextButton", parent)
    btn.Size = UDim2.new(0, 35, 0, 35)
    btn.BackgroundColor3 = bgColor
    btn.Text = ""
    btn.ZIndex = 102
    btn.BackgroundTransparency = 1
    btn.AutoButtonColor = false
    btn:SetAttribute("ColorName", colorName)
    Instance.new("UICorner", btn).CornerRadius = UDim.new(0, 6)
    local stroke = Instance.new("UIStroke", btn)
    stroke.Color = Color3.fromRGB(255, 255, 255)
    stroke.Thickness = 1.5
    stroke.Transparency = 1
    Instance.new("UIScale", btn)
    return btn
end

local function AddColorBtnAnim(btn, glowColor)
    local scale = btn:FindFirstChildOfClass("UIScale")
    btn.MouseEnter:Connect(function()
        local t = (btn:GetAttribute("TargetScale") or 0.85) + 0.08
        TweenService:Create(scale, TweenInfo.new(0.12, Enum.EasingStyle.Sine, Enum.EasingDirection.Out), {Scale = t}):Play()
        if btn.Name ~= "ActiveColorBtn" then
            local s = btn:FindFirstChildOfClass("UIStroke")
            if s then TweenService:Create(s, TweenInfo.new(0.12), {Transparency = 0.5}):Play() end
        end
    end)
    btn.MouseLeave:Connect(function()
        local t = btn:GetAttribute("TargetScale") or 0.85
        TweenService:Create(scale, TweenInfo.new(0.12, Enum.EasingStyle.Sine), {Scale = t}):Play()
        if btn.Name ~= "ActiveColorBtn" then
            local s = btn:FindFirstChildOfClass("UIStroke")
            if s then TweenService:Create(s, TweenInfo.new(0.12), {Transparency = 1}):Play() end
        end
    end)
    btn.MouseButton1Down:Connect(function()
        local t = (btn:GetAttribute("TargetScale") or 0.85) - 0.05
        TweenService:Create(scale, TweenInfo.new(0.08, Enum.EasingStyle.Sine), {Scale = t}):Play()
    end)
    btn.MouseButton1Click:Connect(function()
        ClickSound:Play()
        ApplyColorTheme(glowColor, btn)
        SaveConfig()
    end)
end

-- Performance
local EffectConnection = nil
local function ApplyNoEffects(enabled)
    _G_settings.NoEffects = enabled
    if enabled then
        Lighting.GlobalShadows = false
        for _, v in pairs(Lighting:GetChildren()) do if v:IsA("PostEffect") then v.Enabled = false end end
        local function cleanEffects(v)
            if v:IsA("ParticleEmitter") or v:IsA("Trail") or v:IsA("Smoke") or v:IsA("Fire") or v:IsA("Sparkles") then v:Destroy()
            elseif v:IsA("BasePart") then if v.Parent and not v.Parent:FindFirstChild("Humanoid") then v.Material = Enum.Material.SmoothPlastic end
            elseif (v:IsA("Texture") or v:IsA("Decal")) and v.Name ~= "face" then v:Destroy() end
        end
        for _, v in pairs(workspace:GetDescendants()) do cleanEffects(v) end
        EffectConnection = workspace.DescendantAdded:Connect(function(v) if _G_settings.NoEffects then task.wait(); cleanEffects(v) end end)
    else
        if EffectConnection then EffectConnection:Disconnect() end
    end
end

local PerformanceContent = MakeAccordion(SetP, "PERFORMANCE")
NewBtn("FPS", PerformanceContent, function()
    _G_settings.FPSUnlocked = not _G_settings.FPSUnlocked
    if _G_settings.FPSUnlocked then
        if setfpscap then setfpscap(999) end
        ShowModeMsg("FPS Unlocked")
    else
        if setfpscap then setfpscap(60) end
        ShowModeMsg("FPS Locked to 60")
    end
    Refresh(); SaveConfig()
end)

NewBtn("ShowFPS", PerformanceContent, function()
    _G_settings.ShowFPSCounter = not _G_settings.ShowFPSCounter
    FPSDisplay.Visible = _G_settings.ShowFPSCounter
    Refresh(); SaveConfig()
end)

NewBtn("NoFx", PerformanceContent, function()
    ApplyNoEffects(not _G_settings.NoEffects)
    if _G_settings.NoEffects then ShowModeMsg("Performance Mode ON") else ShowModeMsg("Performance Mode OFF") end
    Refresh(); SaveConfig()
end)

-- FPS Counter
local LastFPSUpdate = tick()
local FrameCount = 0
RunService.RenderStepped:Connect(function()
    FrameCount = FrameCount + 1
    if tick() - LastFPSUpdate >= 0.5 then
        if _G_settings.ShowFPSCounter then FPSDisplay.Text = "FPS: " .. math.floor(FrameCount / (tick() - LastFPSUpdate)) end
        FrameCount = 0; LastFPSUpdate = tick()
    end
end)

local TabData = {
    Farm = {Page = FarmP, CG = FarmCG},
    Player = {Page = PlayerP, CG = PlayerCG},
    Esp = {Page = EspP, CG = EspCG},
    Srvr = {Page = SrvrP, CG = SrvrCG},
    Set = {Page = SetP, CG = SetCG},
    Cred = {Page = CredP, CG = CredCG}
}

local currentActiveTab = nil

local function SwitchTab(activeKey)
    if currentActiveTab == activeKey then return end
    currentActiveTab = activeKey
    OpenSound:Play()
    for key, data in pairs(TabData) do
        local tabBtn = UI_Elements[key]
        if key == activeKey then
            data.CG.Visible = true
            TweenService:Create(data.CG, TweenInfo.new(0.3, Enum.EasingStyle.Sine, Enum.EasingDirection.Out), {GroupTransparency = 0}):Play()
            if tabBtn then
                local uiScale = tabBtn:FindFirstChildOfClass("UIScale")
                tabBtn:SetAttribute("IsActive", true)
                if uiScale then TweenService:Create(uiScale, TweenInfo.new(0.3, Enum.EasingStyle.Sine, Enum.EasingDirection.Out), {Scale = 1.10}):Play() end
                TweenService:Create(tabBtn, TweenInfo.new(0.3, Enum.EasingStyle.Sine, Enum.EasingDirection.Out), {BackgroundTransparency = 1}):Play()
            end
        else
            if data.CG.Visible then
                local fadeTween = TweenService:Create(data.CG, TweenInfo.new(0.3, Enum.EasingStyle.Sine, Enum.EasingDirection.Out), {GroupTransparency = 1})
                fadeTween:Play()
                task.spawn(function()
                    fadeTween.Completed:Wait()
                    if currentActiveTab ~= key then data.CG.Visible = false end
                end)
            end
            if tabBtn then
                local uiScale = tabBtn:FindFirstChildOfClass("UIScale")
                tabBtn:SetAttribute("IsActive", false)
                if uiScale then TweenService:Create(uiScale, TweenInfo.new(0.3, Enum.EasingStyle.Sine, Enum.EasingDirection.Out), {Scale = 1.0}):Play() end
                TweenService:Create(tabBtn, TweenInfo.new(0.3, Enum.EasingStyle.Sine, Enum.EasingDirection.Out), {BackgroundTransparency = 0.90}):Play()
            end
        end
    end
end

for key, _ in pairs(TabData) do
    UI_Elements[key].MouseButton1Click:Connect(function() SwitchTab(key) end)
end
